// 磁力交互
const handleMagneticMove = (event, item) => {
  const rect = item.getBoundingClientRect();
  const centerX = rect.left + rect.width / 2;
  const centerY = rect.top + rect.height / 2;
  const moveX = (event.clientX - centerX) / rect.width * 30;
  const moveY = (event.clientY - centerY) / rect.height * 30;

  gsap.to(item, {
    x: moveX,
    y: moveY,
    rotateX: moveY * -0.5,
    rotateY: moveX * 0.5,
    duration: 0.3,
    ease: 'power2.out'
  });
};

// 入场动画
gsap.from(gridItems, {
  y: 100,
  opacity: 0,
  scale: 0.8,
  duration: 0.8,
  ease: 'back.out(1.5)',
  stagger: { each: 0.08, from: 'center' },
  scrollTrigger: {
    trigger: gridItems[0],
    start: 'top 85%',
    toggleActions: 'play none none reverse'
  }
});

    }

    container.appendChild(particle);

    gsap.to(particle, {
      y: '-=100',
      x: `+=${Math.random() * 100 - 50}`,
      opacity: 0,
      duration: Math.random() * 4 + 3,
      repeat: -1,
      delay: Math.random() * 2,
      ease: 'none'
    });

}

## 💡 创作建议

**让网站更专业：**
1. ✅ 首页要有记忆点，第一屏足够震撼
2. ✅ 使用 GSAP 的 `timeline` 组合复杂动画
3. ✅ 添加丰富的微交互（悬停、点击、拖拽）
4. ✅ 使用 `stagger` 实现错峰动画
5. ✅ 添加过渡效果，避免生硬切换
6. ✅ 霓虹光效适度，避免视觉疲劳
7. ✅ 3D 效果自然，避免眩晕
8. ✅ 磁力交互流畅，响应及时

**避免常见错误：**
1. ❌ 不要过度使用动画，影响性能（控制同时运行的动画数量）
2. ❌ 不要让动画遮挡内容（动画元素设置 z-index: -1 或 pointer-events: none）
3. ❌ 不要忘记响应式适配（使用 media queries 和 clamp()）
4. ❌ 不要使用过大的图片影响加载（压缩图片，使用 WebP 格式）
5. ❌ 不要忽略移动端体验（触摸事件、手势交互、字体大小）
6. ❌ 不要过度使用霓虹光效（控制发光范围和强度）
7. ❌ 不要让 3D 效果过于剧烈（限制旋转角度和位移距离）
8. ❌ 不要忽略无障碍访问（aria-label、alt 文本、键盘导航）
9. ❌ 不要使用固定的硬编码数值（使用 CSS 变量和相对单位）
10. ❌ 不要忘记清理动画实例（页面卸载时 kill 动画）

**动画节奏控制：**
- Hero 区域：慢速震撼（2-3 秒）
- 特性卡片：中速交互（0.5-1 秒）
- 数据展示：中等节奏（1-2 秒）
- 滚动动画：跟随滚动（scrub: 1）
- 微交互：快速响应（0.2-0.4 秒）

### 性能优化补充说明

**图片优化：**
- 使用 WebP 格式（比 JPEG 小 30%）
- 添加 `loading="lazy"` 懒加载属性
- 使用 `srcset` 提供不同分辨率版本
- 压缩图片质量（85% 质量足够）
- 使用 CDN 加速图片加载

**动画优化：**
- 使用 `will-change: transform, opacity` 提示浏览器优化
- 避免在 scroll 事件中执行重排/重绘
- 使用 `transform: translateZ(0)` 开启 GPU 加速
- 减少 box-shadow 的模糊半径（> 50px 会影响性能）
- 使用 `requestAnimationFrame` 优化动画帧
- 粒子系统复用 DOM 元素（销毁后重用，不是新建）

**代码优化：**
- 使用 CSS 变量减少重复代码
- 合并相似的动画函数
- 使用事件委托减少事件监听器数量
- 避免在循环中创建函数（闭包问题）
- 及时清理不需要的动画实例
- 使用 `debounce` 和 `throttle` 优化频繁触发的事件
