    end: 'bottom 60%',

function updateActiveNav(id) {
  document.querySelectorAll('.nav-link').forEach(link => {
    link.classList.remove('active');
    if (link.getAttribute('href') === `#${id}`) {
      link.classList.add('active');
    }
  });
}

// 磁力按钮效果（Logo）
initMagneticButton('.logo-magnetic');

// 平滑滚动导航
document.querySelectorAll('.nav-link').forEach(link => {
  link.addEventListener('click', (e) => {
    e.preventDefault();
    const targetId = link.getAttribute('href');
    const target = document.querySelector(targetId);
    if (target) {
      const offset = 80;
      const targetPosition = target.getBoundingClientRect().top + window.pageYOffset - offset;
      gsap.to(window, {
        duration: 1.2,
        scrollTo: targetPosition,
        ease: 'power3.inOut'
      });
    }
  });
});

// 极光背景
function initAuroraBorealis() {
  const auroraCanvas = document.getElementById('aurora-canvas');
  const ctx = auroraCanvas.getContext('2d');
  
  auroraCanvas.width = window.innerWidth;
  auroraCanvas.height = window.innerHeight;
  
  const waves = [];
  const waveCount = 5;
  
  for (let i = 0; i < waveCount; i++) {
    waves.push({
      amplitude: 50 + i * 20,
      frequency: 0.003 + i * 0.001,
      phase: i * 0.5,
      speed: 0.02 + i * 0.01,
      color: `hsla(${200 + i * 20}, 80%, 60%, ${0.3 - i * 0.05})`
    });
  }
  
  let time = 0;
  
  function drawAurora() {
    ctx.clearRect(0, 0, auroraCanvas.width, auroraCanvas.height);
    
    waves.forEach((wave, i) => {
      ctx.beginPath();
      ctx.moveTo(0, auroraCanvas.height);
      
      for (let x = 0; x <= auroraCanvas.width; x += 5) {
        const y = Math.sin(x * wave.frequency + time + wave.phase) * wave.amplitude +
                  Math.sin(x * wave.frequency * 2 + time * 0.5) * wave.amplitude * 0.5 +
                  auroraCanvas.height * 0.4 + i * 100;
        ctx.lineTo(x, y);
      }
      
      ctx.lineTo(auroraCanvas.width, auroraCanvas.height);
      ctx.closePath();
      
      ctx.fillStyle = wave.color;
      ctx.fill();
    });
    
    time += 0.02;
    requestAnimationFrame(drawAurora);
  }
  
  drawAurora();
}

// 海洋波浪
function initOceanWaves() {
  const oceanCanvas = document.getElementById('ocean-canvas');
  const ctx = oceanCanvas.getContext('2d');
  
  oceanCanvas.width = window.innerWidth;
  oceanCanvas.height = 200;
  
  const waves = [];
  const waveCount = 3;
  
  for (let i = 0; i < waveCount; i++) {
    waves.push({
      y: 100 + i * 30,
      amplitude: 20 + i * 10,
      frequency: 0.02 + i * 0.01,
      phase: i * 2,
      speed: 0.03 + i * 0.02,
      color: `rgba(14, 165, 233, ${0.2 - i * 0.05})`
    });
  }
  
  let time = 0;
  
  function drawOcean() {
    ctx.clearRect(0, 0, oceanCanvas.width, oceanCanvas.height);
    
    waves.forEach(wave => {
      ctx.beginPath();
      ctx.moveTo(0, oceanCanvas.height);
      
      for (let x = 0; x <= oceanCanvas.width; x += 5) {
        const y = Math.sin(x * wave.frequency + time + wave.phase) * wave.amplitude + wave.y;
        ctx.lineTo(x, y);
      }
      
      ctx.lineTo(oceanCanvas.width, oceanCanvas.height);
      ctx.closePath();
      
      ctx.fillStyle = wave.color;
      ctx.fill();
    });
    
    time += 0.02;
    requestAnimationFrame(drawOcean);
  }
  
  drawOcean();
}

// 文字波浪变形
function initTextWaveMorph(selector) {
  const element = document.querySelector(selector);
  if (!element) return;
  
  const text = element.textContent;
  element.innerHTML = '';
  
  text.split('').forEach((char, i) => {
    const span = document.createElement('span');
    span.textContent = char === ' ' ? '\u00A0' : char;
    span.style.display = 'inline-block';
    element.appendChild(span);
  });
  
  const chars = element.querySelectorAll('span');
  
  chars.forEach((char, i) => {
    gsap.to(char, {
      y: -10,
      duration: 2 + i * 0.1,
      repeat: -1,
      yoyo: true,
      ease: 'sine.inOut',
      delay: i * 0.05
    });
  });
}

// 鼠标移动视差
function initMouseParallax() {
  const heroSection = document.querySelector('.hero-section');
  
  heroSection.addEventListener('mousemove', (e) => {
    const x = (e.clientX / window.innerWidth - 0.5) * 30;
    const y = (e.clientY / window.innerHeight - 0.5) * 30;
    
    gsap.to('.hero-bg-layer-1', {
      x: x * 0.5,
      y: y * 0.5,
      duration: 1,
      ease: 'power2.out'
    });
    
    gsap.to('.hero-bg-layer-2', {
      x: x * 0.3,
      y: y * 0.3,
      duration: 1,
      ease: 'power2.out'
    });
    
    gsap.to('.hero-content', {
      x: x * 0.1,
      y: y * 0.1,
      duration: 1,
      ease: 'power2.out'
    });
  });
}

**实现代码：**
```javascript
// 热门目的地初始化
function initDestinations() {
  const track = document.querySelector('.destinations-track');
  const cards = gsap.utils.toArray('.destination-card');
  
  // 横向滚动
  gsap.to(track, {
    x: () => -(track.scrollWidth - window.innerWidth),
    ease: 'none',
    scrollTrigger: {
      trigger: track,
      start: 'top top',
      end: () => '+=' + (track.scrollWidth - window.innerWidth),
      pin: true,
      scrub: 1,
      anticipatePin: 1
    }
  });
  
  // 卡片入场动画
  cards.forEach((card, i) => {
    gsap.from(card, {
      y: 100,
      opacity: 0,
      rotateY: -20,
      scale: 0.8,
      duration: 0.8,
      ease: 'power3.out',
      scrollTrigger: {
        trigger: card,
        start: 'left 80%',
        end: 'right 20%',
        containerAnimation: gsap.to(track, { x: () => -(track.scrollWidth - window.innerWidth) }),
        toggleActions: 'play none none reverse'
      }
    });
    
    // 图片视差
    const image = card.querySelector('.destination-image img');
    gsap.to(image, {
      y: -50,
      ease: 'none',
      scrollTrigger: {
        trigger: card,
        start: 'top bottom',
        end: 'bottom top',
        containerAnimation: gsap.to(track, { x: () => -(track.scrollWidth - window.innerWidth) }),
        scrub: true
      }
    });
    
    // 悬停效果
    card.addEventListener('mouseenter', () => {
      gsap.to(card, {
        scale: 1.05,
        y: -20,
        boxShadow: '0 30px 60px rgba(0, 0, 0, 0.3)',
        duration: 0.4,
        ease: 'power2.out'
      });
      
      gsap.to(image, {
        scale: 1.1,
        duration: 0.4,
        ease: 'power2.out'
      });
    });
    
    card.addEventListener('mouseleave', () => {
      gsap.to(card, {
        scale: 1,
        y: 0,
        boxShadow: '0 20px 40px rgba(0, 0, 0, 0.2)',
        duration: 0.4,
        ease: 'power2.out'
      });
      
      gsap.to(image, {
        scale: 1,
        duration: 0.4,
        ease: 'power2.out'
      });
    });
  });
  
  // 滑动指示器
  const indicator = document.querySelector('.scroll-indicator-bar');
  const indicatorProgress = indicator.querySelector('.indicator-progress');
  
  ScrollTrigger.create({
    trigger: track,
    start: 'top top',
    end: () => '+=' + (track.scrollWidth - window.innerWidth),
    scrub: true,
    onUpdate: (self) => {
      gsap.to(indicatorProgress, {
        width: self.progress * 100 + '%',
        duration: 0.1,
        ease: 'none'
      });
    }
  });
}
```

function initRotationCarousel() {
  const carousel = document.querySelector('.experience-carousel');
  const items = gsap.utils.toArray('.experience-item');
  const itemAngle = 360 / items.length;
  const radius = 400;
  
  items.forEach((item, i) => {
    const angle = itemAngle * i;
    gsap.set(item, {
      position: 'absolute',
      top: '50%',
      left: '50%',
      width: '300px',
      height: '400px',
      margin: '-200px 0 0 -150px',
      rotateY: angle,
      translateZ: radius
    });
  });
  
  let currentRotation = 0;
  
  function rotateCarousel(direction) {
    currentRotation += direction * itemAngle;
    gsap.to(items, {
      rotateY: (i) => i * itemAngle + currentRotation,
      duration: 0.8,
      ease: 'power2.out'
    });
  }
  
  // 自动轮播
  let autoRotate = setInterval(() => rotateCarousel(-1), 3000);
  
  carousel.addEventListener('mouseenter', () => clearInterval(autoRotate));
  carousel.addEventListener('mouseleave', () => {
    autoRotate = setInterval(() => rotateCarousel(-1), 3000);
  });
  
  // 左右控制
  carousel.querySelector('.carousel-prev').addEventListener('click', () => rotateCarousel(1));
  carousel.querySelector('.carousel-next').addEventListener('click', () => rotateCarousel(-1));
}

### 📍 区域 9：返回顶部按钮
**使用组件：** `ScrollReveal`, `ScrollProgress`, `ScrollShapeElastic`

**动画效果：**
- 滚动超过500px时显示
- 平滑滚动到顶部
- 悬停时弹性形状动画
- 进度圆环指示器

**实现代码：**
```javascript
// 返回顶部按钮初始化
function initBackToTop() {
  const backToTop = document.getElementById('backToTop');
  const progressCircle = backToTop.querySelector('.progress-circle');
  const circumference = 2 * Math.PI * 22;
  
  // 初始隐藏
  gsap.set(backToTop, { opacity: 0, visibility: 'hidden' });
  
  // 滚动监听
  window.addEventListener('scroll', () => {
    if (window.scrollY > 500) {
      gsap.to(backToTop, {
        opacity: 1,
        visibility: 'visible',
        duration: 0.3,
        ease: 'power2.out'
      });
    } else {
      gsap.to(backToTop, {
        opacity: 0,
        visibility: 'hidden',
        duration: 0.3,
        ease: 'power2.out'
      });
    }
    
    // 更新进度
    const scrollProgress = window.scrollY / (document.documentElement.scrollHeight - window.innerHeight);
    const offset = circumference * (1 - scrollProgress);
    gsap.set(progressCircle, {
      strokeDashoffset: offset
    });
  });
  
  // 点击返回顶部
  backToTop.addEventListener('click', () => {
    gsap.to(window, {
      duration: 1.2,
      scrollTo: 0,
      ease: 'power3.inOut'
    });
  });
  
  // 悬停弹性效果
  backToTop.addEventListener('mouseenter', () => {
    gsap.to(backToTop, {
      scale: 1.1,
      rotate: 45,
      duration: 0.4,
      ease: 'elastic.out(1, 0.5)'
    });
  });
  
  backToTop.addEventListener('mouseleave', () => {
    gsap.to(backToTop, {
      scale: 1,
      rotate: 0,
      duration: 0.4,
      ease: 'power2.out'
    });
  });
}
```
