// 极光背景
function initAuroraBackground() {
  const aurora = document.getElementById('aurora-canvas');
  const ctx = aurora.getContext('2d');
  
  let time = 0;
  function animateAurora() {
    ctx.clearRect(0, 0, aurora.width, aurora.height);
    
    // 绘制多层波浪
    for (let i = 0; i < 5; i++) {
      ctx.beginPath();
      ctx.moveTo(0, aurora.height);
      
      for (let x = 0; x <= aurora.width; x += 10) {
        const y = Math.sin(x * 0.003 + time + i) * 50 + 
                  Math.sin(x * 0.006 + time * 0.5 + i) * 30;
        ctx.lineTo(x, y + 200 + i * 100);
      }
      
      ctx.lineTo(aurora.width, aurora.height);
      ctx.closePath();
      
      const gradient = ctx.createLinearGradient(0, 0, 0, aurora.height);
      gradient.addColorStop(0, `hsla(${200 + i * 10}, 80%, 60%, ${0.1 - i * 0.02})`);
      gradient.addColorStop(1, 'transparent');
      
      ctx.fillStyle = gradient;
      ctx.fill();
    }
    
    time += 0.02;
    requestAnimationFrame(animateAurora);
  }
  
  animateAurora();
}

function initFeatureCards() {
  const cards = gsap.utils.toArray('.feature-card');
  
  // 入场动画
  gsap.from(cards, {
    y: 100,
    opacity: 0,
    scale: 0.8,
    duration: 0.8,
    ease: 'back.out(1.5)',
    stagger: { each: 0.1, from: 'center' },
    scrollTrigger: {
      trigger: '.features-section',
      start: 'top 85%',
      toggleActions: 'play none none reverse'
    }
  });

  // 磁力交互
  cards.forEach(card => {
    const neonAura = card.querySelector('.neon-aura');
    const icon = card.querySelector('.feature-icon');
    
    card.addEventListener('mousemove', (e) => {
      const rect = card.getBoundingClientRect();
      const centerX = rect.left + rect.width / 2;
      const centerY = rect.top + rect.height / 2;
      const moveX = (e.clientX - centerX) / rect.width * 30;
      const moveY = (e.clientY - centerY) / rect.height * 30;

        y: moveY,
        rotateX: moveY * -0.5,
        rotateY: moveX * 0.5,
        scale: 1.05,
        duration: 0.3,
        ease: 'power2.out'

        opacity: 0.6,
        scale: 1.3,
        duration: 0.3
      });

      // 图标旋转
      gsap.to(icon, {
        rotate: 45,
        scale: 1.2,
        duration: 0.3,
        ease: 'back.out(1.5)'
      });
    });

    card.addEventListener('mouseleave', () => {
      gsap.to(card, {
        x: 0,
        y: 0,
        rotateX: 0,
        rotateY: 0,
        scale: 1,
        duration: 0.5,
        ease: 'power2.out'
      });

      gsap.to(neonAura, {
        opacity: 0.3,
        scale: 1,
        duration: 0.5
      });

      gsap.to(icon, {
        rotate: 0,
        scale: 1,
        duration: 0.5,
        ease: 'power2.out'

    trigger: '.features-section',
    start: 'top 80%',
    end: 'bottom 20%',

      const progress = self.progress;
      cards.forEach(card => {
        const aura = card.querySelector('.neon-aura');
        gsap.to(aura, {
          opacity: 0.2 + progress * 0.4,
          scale: 1 + progress * 0.3
        });
      });
    }
  });
}

// 性能参数初始化
function initPerformanceSection() {
  // 进度圆环动画
  const progressCircles = gsap.utils.toArray('.progress-circle');
  progressCircles.forEach(circle => {
    const value = parseFloat(circle.dataset.value);
    const circumference = 2 * Math.PI * 45; // r=45
    
    circle.querySelector('.progress-path').style.strokeDasharray = circumference;
    circle.querySelector('.progress-path').style.strokeDashoffset = circumference;
    
    gsap.to(circle.querySelector('.progress-path'), {
      strokeDashoffset: circumference * (1 - value / 100),
      duration: 2,
      ease: 'power3.out',
      scrollTrigger: {
        trigger: circle,
        start: 'top 80%',
        toggleActions: 'play none none reverse'
      }
    });
  });

  // 数字计数动画
  initNumberCounters('.stat-number', 2);

  // MotionPath数据流动画
  gsap.to('.data-particle', {
    motionPath: {
      path: '#data-path-1',
      align: '#data-path-1',
      alignOrigin: [0.5, 0.5]
    },
    duration: 3,
    repeat: -1,
    ease: 'none'
  });

  // 赛博脉冲
  initCyberPulse();
}

// 数字计数动画
function initNumberCounters(selector, duration = 2) {
  const counters = gsap.utils.toArray(selector);
  
  counters.forEach(counter => {
    const target = parseFloat(counter.dataset.target);
    
    gsap.to(counter, {
      innerHTML: target,
      duration: duration,
      ease: 'power2.out',
      snap: { innerHTML: 1 },
      scrollTrigger: {
        trigger: counter,
        start: 'top 85%',
        toggleActions: 'play none none reverse'
      },
      onUpdate: function() {
        counter.innerHTML = Math.ceil(this.targets()[0].innerHTML)
          .toLocaleString('en-US');
      }
    });
  });
}

// 赛博脉冲效果
function initCyberPulse() {
  const pulses = gsap.utils.toArray('.cyber-pulse');
  
  pulses.forEach((pulse, i) => {
    gsap.to(pulse, {
      scale: 2,
      opacity: 0,
      duration: 2,
      repeat: -1,
      delay: i * 0.5,
      ease: 'power2.out'
    });
  });
}

// 图片对比功能
function initImageComparison() {
  const comparisons = gsap.utils.toArray('.image-comparison');
  
  comparisons.forEach(comp => {
    const slider = comp.querySelector('.comparison-slider');
    const overlay = comp.querySelector('.comparison-overlay');
    let isDragging = false;

    const handleMove = (e) => {
      if (!isDragging) return;
      const rect = comp.getBoundingClientRect();
      const x = e.clientX || e.touches[0].clientX;
      const percentage = Math.max(0, Math.min(100, (x - rect.left) / rect.width * 100));

      gsap.to(slider, {
        left: percentage + '%',
        duration: 0.1,
        ease: 'none'
      });

      gsap.to(overlay, {
        width: percentage + '%',
        duration: 0.1,
        ease: 'none'
      });
    };

    slider.addEventListener('mousedown', () => isDragging = true);
    slider.addEventListener('touchstart', () => isDragging = true);
    document.addEventListener('mouseup', () => isDragging = false);
    document.addEventListener('touchend', () => isDragging = false);
    document.addEventListener('mousemove', handleMove);
    document.addEventListener('touchmove', handleMove);
  });

// 返回顶部按钮初始化
function initBackToTop() {
  const backToTop = document.getElementById('backToTop');
  const progressCircle = backToTop.querySelector('.progress-circle');
  const circumference = 2 * Math.PI * 22; // r=22

  // 初始隐藏
  gsap.set(backToTop, { opacity: 0, visibility: 'hidden' });

  // 滚动监听显示/隐藏
  window.addEventListener('scroll', () => {
    if (window.scrollY > 500) {
      gsap.to(backToTop, {
        opacity: 1,
        visibility: 'visible',
        duration: 0.3,
        ease: 'power2.out'
      });
    } else {
      gsap.to(backToTop, {
        opacity: 0,
        visibility: 'hidden',
        duration: 0.3,
        ease: 'power2.out'
      });
    }

    // 更新进度圆环
    const scrollProgress = window.scrollY / (document.documentElement.scrollHeight - window.innerHeight);
    const offset = circumference * (1 - scrollProgress);
    gsap.set(progressCircle, {
      strokeDashoffset: offset
    });
  });

  // 点击返回顶部
  backToTop.addEventListener('click', () => {
    gsap.to(window, {
      duration: 1,
      scrollTo: 0,
      ease: 'power3.inOut'
    });
  });

  // 悬停动画
  backToTop.addEventListener('mouseenter', () => {
    gsap.to(backToTop, {
      rotate: 180,
      scale: 1.1,
      boxShadow: '0 0 30px rgba(59, 130, 246, 0.8)',
      duration: 0.4,
      ease: 'back.out(1.5)'
    });
  });

  backToTop.addEventListener('mouseleave', () => {
    gsap.to(backToTop, {
      rotate: 0,
      scale: 1,
      boxShadow: '0 0 15px rgba(59, 130, 246, 0.5)',
      duration: 0.4,
      ease: 'power2.out'
    });
  });

  // 磁力效果
  initMagneticButton(backToTop);
}

## 🚀 滚动主题切换功能

### 实现思路
参考 `ScrollThemeSwitch.vue` 的思路，创建全局主题变量，滚动时动态切换。

```javascript
// 全局主题管理
class ThemeManager {
  constructor() {
    this.currentTheme = 'dark';
    this.themes = {
      dark: {
        primary: '#3b82f6',
        dark: '#0f172a',
        light: '#ffffff',
        accent: '#8b5cf6',
        gradient: 'linear-gradient(135deg, #3b82f6 0%, #8b5cf6 100%)'
      },
      blue: {
        primary: '#06b6d4',
        dark: '#0c4a6e',
        light: '#ffffff',
        accent: '#3b82f6',
        gradient: 'linear-gradient(135deg, #06b6d4 0%, #3b82f6 100%)'
      },
      purple: {
        primary: '#8b5cf6',
        dark: '#2e1065',
        light: '#ffffff',
        accent: '#ec4899',
        gradient: 'linear-gradient(135deg, #8b5cf6 0%, #ec4899 100%)'
      }
    };
    
    this.init();
  }

  init() {
    // 根据滚动位置切换主题
    const themeSections = [
      { trigger: '.hero-section', theme: 'dark' },
      { trigger: '.features-section', theme: 'blue' },
      { trigger: '.performance-section', theme: 'purple' },
      { trigger: '.gallery-section', theme: 'dark' },
      { trigger: '.experience-section', theme: 'blue' }
    ];

    themeSections.forEach(section => {
      ScrollTrigger.create({
        trigger: section.trigger,
        start: 'top center',
        end: 'bottom center',
        onEnter: () => this.setTheme(section.theme),
        onEnterBack: () => this.setTheme(section.theme)
      });
    });
  }

  setTheme(themeName) {
    if (this.currentTheme === themeName) return;
    this.currentTheme = themeName;
    
    const theme = this.themes[themeName];
    const root = document.documentElement;

    // 更新CSS变量
    gsap.to(root, {
      '--color-primary': theme.primary,
      '--color-dark': theme.dark,
      '--color-accent': theme.accent,
      '--gradient-primary': theme.gradient,
      duration: 0.5,
      ease: 'power2.out'
    });
  }
}

// 初始化
const themeManager = new ThemeManager();
```
