## 🏆 顶级动画组件集成方案

### 1️⃣ ScrollCrystalPrism - 水晶棱镜效果

**应用场景：** 钻石系列展示、高端定制页面

**核心功能：**
- 棱镜3D旋转与光束反射效果
- 光谱色带动态展示
- Canvas粒子系统模拟星尘效果
- 鼠标悬停触发光束强度变化

**集成代码：**
```javascript
/**
 * ScrollCrystalPrism 水晶棱镜效果集成
 * 完美展现钻石火彩与光学特性
 */
function initScrollCrystalPrism() {
  const prisms = document.querySelectorAll('.crystal-prism');
  
  prisms.forEach(prism => {
    // 创建棱镜层
    const prismLayers = [];
    for (let i = 0; i < 6; i++) {
      const layer = document.createElement('div');
      layer.className = `scp-layer scp-layer-${i}`;
      layer.style.cssText = `
        position: absolute;
        inset: 0;
        background: linear-gradient(${45 + i * 30}deg, rgba(212, 175, 55, ${0.1 + i * 0.05}), transparent);
        pointer-events: none;
      `;
      prism.appendChild(layer);
      prismLayers.push(layer);
    }
    
    // 持续旋转
    gsap.to(prism, {
      rotateY: 360,
      duration: 10,
      repeat: -1,
      ease: 'none'
    });
    
    // 光线散射动画
    const lightRays = [];
    for (let i = 0; i < 8; i++) {
      const ray = document.createElement('div');
      ray.className = `scp-light-ray scp-light-ray-${i}`;
      ray.style.cssText = `
        position: absolute;
        top: 50%;
        left: 50%;
        width: 2px;
        height: 200px;
        background: linear-gradient(to top, rgba(212, 175, 55, 0.8), transparent);
        transform-origin: bottom center;
        transform: rotate(${i * 45}deg);
        pointer-events: none;
        opacity: 0;
      `;
      prism.appendChild(ray);
      lightRays.push(ray);
    }
    
    // 鼠标悬停交互
    prism.addEventListener('mouseenter', () => {
      gsap.to(prism, {
        scale: 1.15,
        rotateY: 20,
        duration: 0.4,
        ease: 'power2.out'
      });
      
      // 光线强度增强
      lightRays.forEach((ray, i) => {
        gsap.to(ray, {
          opacity: 1,
          scaleY: 1.5,
          duration: 0.3,
          delay: i * 0.05,
          ease: 'power2.out'
        });
      });
      
      // 棱镜层动画
      prismLayers.forEach((layer, index) => {
        gsap.to(layer, {
          opacity: 0.6,
          duration: 0.5,
          delay: index * 0.1,
          ease: 'power2.out'
        });
      });
    });
    
    prism.addEventListener('mouseleave', () => {
      gsap.to(prism, {
        scale: 1,
        rotateY: 0,
        duration: 0.4,
        ease: 'power2.inOut'
      });
      
      // 光线减弱
      lightRays.forEach(ray => {
        gsap.to(ray, {
          opacity: 0,
          scaleY: 1,
          duration: 0.3,
          ease: 'power2.inOut'
        });
      });
      
      // 棱镜层淡出
      prismLayers.forEach(layer => {
        gsap.to(layer, {
          opacity: 0,
          duration: 0.5,
          ease: 'power2.inOut'
        });
      });
    });
    
    // 滚动控制
    ScrollTrigger.create({
      trigger: prism,
      start: 'top 70%',
      end: 'bottom 30%',
      scrub: 1,
      onUpdate: (self) => {
        const progress = self.progress;
        gsap.to(lightRays, {
          opacity: 0.5 + progress * 0.5,
          duration: 0.1
        });
      }
    });
  });
}
```

### 2️⃣ ScrollNeonPrismCards - 霓虹棱镜卡片

**应用场景：** 夜间系列珠宝、时尚珠宝、活动专题

**核心功能：**
- 霓虹光晕与氛围光环
- 棱镜反射与折射层
- 能量条动画
- 极光背景动态
- 40个粒子星云效果

**集成代码：**
```javascript
/**
 * ScrollNeonPrismCards 霓虹棱镜卡片集成
 * 营造高端科技感与奢华氛围
 */
function initScrollNeonPrismCards() {
  const neonCards = document.querySelectorAll('.neon-prism-card');
  
  neonCards.forEach(card => {
    // 霓虹光晕
    const neonGlow = document.createElement('div');
    neonGlow.className = 'npr-neon-glow';
    neonGlow.style.cssText = `
      position: absolute;
      inset: -20px;
      background: radial-gradient(circle, rgba(212, 175, 55, 0.3), transparent 70%);
      opacity: 0;
      transition: opacity 0.4s ease;
      pointer-events: none;
    `;
    card.appendChild(neonGlow);
    
    // 能量环系统
    const energyRings = [];
    for (let i = 0; i < 4; i++) {
      const ring = document.createElement('div');
      ring.className = `npr-energy-ring npr-energy-ring-${i}`;
      ring.style.cssText = `
        position: absolute;
        top: 50%;
        left: 50%;
        transform: translate(-50%, -50%);
        width: ${120 + i * 40}px;
        height: ${120 + i * 40}px;
        border: 2px solid rgba(212, 175, 55, ${0.4 - i * 0.1});
        border-radius: 50%;
        pointer-events: none;
      `;
      card.appendChild(ring);
      energyRings.push(ring);
    }
    
    // 环旋转动画
    energyRings.forEach((ring, index) => {
      gsap.to(ring, {
        rotation: index % 2 === 0 ? 360 : -360,
        duration: 10 + index * 2,
        repeat: -1,
        ease: 'none'
      });
    });
    
    // 粒子系统
    const particleCount = 40;
    const particles = [];
    for (let i = 0; i < particleCount; i++) {
      const particle = document.createElement('div');
      particle.className = 'npr-particle';
      
      const angle = (i / particleCount) * Math.PI * 2;
      const radius = 80 + Math.random() * 40;
      
      particle.style.cssText = `
        position: absolute;
        top: 50%;
        left: 50%;
        width: 4px;
        height: 4px;
        background: var(--color-gold-500);
        border-radius: 50%;
        pointer-events: none;
        box-shadow: 0 0 10px var(--color-gold-500);
      `;
      
      card.appendChild(particle);
      particles.push({
        element: particle,
        angle,
        radius,
        speed: 0.01 + Math.random() * 0.02
      });
    }
    
    // 粒子动画循环
    let animationFrame;
    function animateParticles() {
      particles.forEach(particle => {
        particle.angle += particle.speed;
        
        const x = Math.cos(particle.angle) * particle.radius;
        const y = Math.sin(particle.angle) * particle.radius;
        
        particle.element.style.transform = `translate(calc(-50% + ${x}px), calc(-50% + ${y}px))`;
      });
      
      animationFrame = requestAnimationFrame(animateParticles);
    }
    
    animateParticles();
    
    // 鼠标交互
    card.addEventListener('mousemove', (e) => {
      const rect = card.getBoundingClientRect();
      const x = ((e.clientX - rect.left) / rect.width - 0.5) * 15;
      const y = ((e.clientY - rect.top) / rect.height - 0.5) * 15;
      
      gsap.to(card, {
        rotateY: x,
        rotateX: -y,
        duration: 0.4,
        ease: 'power2.out'
      });
      
      gsap.to(neonGlow, {
        opacity: 0.8,
        scale: 1.2,
        duration: 0.4
      });
    });
    
    card.addEventListener('mouseleave', () => {
      gsap.to(card, {
        rotateY: 0,
        rotateX: 0,
        duration: 0.5,
        ease: 'elastic.out(1, 0.5)'
      });
      
      gsap.to(neonGlow, {
        opacity: 0,
        scale: 1,
        duration: 0.4
      });
    });
    
    // 滚动控制
    ScrollTrigger.create({
      trigger: card,
      start: 'top 70%',
      end: 'bottom 30%',
      onUpdate: (self) => {
        const progress = self.progress;
        
        // 环旋转速度加快
        energyRings.forEach(ring => {
          const tweens = gsap.getTweensOf(ring);
          tweens.forEach(tween => {
            tween.timeScale(1 + progress * 3);
          });
        });
        
        // 颜色变化
        const hue = 45 + progress * 20;
        card.style.borderColor = `hsl(${hue}, 80%, 50%)`;
      }
    });
  });
}
```

/**
 * ScrollHologramGallery 全息投影画廊集成
 * 展现科技感与未来感的珠宝展示
 */
function initScrollHologramGallery() {
  const hologramItems = document.querySelectorAll('.hologram-item');
  
  hologramItems.forEach(item => {
    const image = item.querySelector('.hologram-image');
    const hologramImage = item.querySelector('.shg-hologram-image-160');
    
    // 扫描线动画
    const scanline = document.createElement('div');
    scanline.className = 'shg-scanline-160';
    scanline.style.cssText = `
      position: absolute;
      inset: 0;
      background: linear-gradient(180deg, transparent 49%, rgba(212, 175, 55, 0.1) 50%, transparent 51%);
      background-size: 100% 8px;
      pointer-events: none;
    `;
    hologramImage.appendChild(scanline);
    
    gsap.to(scanline, {
      backgroundPositionY: '100%',
      duration: 3,
      repeat: -1,
      ease: 'none'
    });
    
    // 噪点效果
    const noise = document.createElement('div');
    noise.className = 'shg-noise-160';
    noise.style.cssText = `
      position: absolute;
      inset: 0;
      background: url('data:image/svg+xml,<svg viewBox="0 0 200 200" xmlns="http://www.w3.org/2000/svg"><filter id="noise"><feTurbulence type="fractalNoise" baseFrequency="0.65" numOctaves="3" stitchTiles="stitch"/></filter><rect width="100%" height="100%" filter="url(%23noise)" opacity="0.1"/></svg>');
      opacity: 0.3;
      pointer-events: none;
    `;
    hologramImage.appendChild(noise);
    
    gsap.to(noise, {
      opacity: 0.3 + Math.random() * 0.4,
      duration: 0.1,
      repeat: -1,
      yoyo: true
    });
    
    // 基底环旋转
    const baseRing = item.querySelector('.base-ring');
    if (baseRing) {
      gsap.to(baseRing, {
        rotation: 360,
        duration: 20,
        repeat: -1,
        ease: 'none'
      });
    }
    
    // 全息投影3D交互
    image.addEventListener('mousemove', (e) => {
      const rect = image.getBoundingClientRect();
      const x = ((e.clientX - rect.left) / rect.width) * 2 - 1;
      const y = ((e.clientY - rect.top) / rect.height) * 2 - 1;
      
      gsap.to(hologramImage, {
        rotateY: x * 15,
        rotateX: -y * 15,
        duration: 0.3,
        ease: 'power2.out'
      });
      
      // 光束动态效果
      const beam = item.querySelector('.shg-beam-160');
      if (beam) {
        gsap.to(beam, {
          opacity: 0.3 + Math.abs(x) * 0.4,
          scaleX: 1 + Math.abs(x) * 0.3,
          duration: 0.3
        });
      }
    });
    
    image.addEventListener('mouseleave', () => {
      gsap.to(hologramImage, {
        rotateY: 0,
        rotateX: 0,
        duration: 0.5,
        ease: 'power2.inOut'
      });
    });
    
    // 状态指示灯
    const statusIndicator = item.querySelector('.status-indicator');
    if (statusIndicator) {
      gsap.to(statusIndicator, {
        opacity: 0.5,
        scale: 0.9,
        duration: 2,
        repeat: -1,
        yoyo: true,
        ease: 'power2.inOut'
      });
    }
  });
}

### 4️⃣ ScrollAuroraField - 极光领域背景

**应用场景：** Hero区域、品牌故事背景、产品详情页

**核心功能：**
- Canvas绘制极光波浪
- 多层颜色渐变与透明度变化
- 滚动控制极光强度
- 卡片悬浮光效

**集成代码：**
```javascript
/**
 * ScrollAuroraField 极光领域背景集成
 * 营造神秘高端的视觉氛围
 */
function initScrollAuroraField() {
  const canvas = document.getElementById('aurora-canvas');
  if (!canvas) return;
  
  const ctx = canvas.getContext('2d');
  let time = 0;
  let animationFrame;
  
  // 设置Canvas尺寸
  function resizeCanvas() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
  }
  resizeCanvas();
  window.addEventListener('resize', resizeCanvas);
  
  // 极光波浪参数
  const auroraWaves = Array.from({ length: 5 }, (_, i) => ({
    x: 0,
    y: canvas.height * (0.3 + i * 0.15),
    amplitude: 50 + i * 20,
    frequency: 0.002 + i * 0.001,
    speed: 0.01 + i * 0.005,
    phase: i * Math.PI * 0.5,
    color: `hsla(${280 + i * 20}, 70%, 50%, 0.3)`
  }));
  
  // 极光绘制函数
  function drawAurora() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // 绘制背景渐变
    const gradient = ctx.createLinearGradient(0, 0, 0, canvas.height);
    gradient.addColorStop(0, 'rgba(13, 13, 13, 0.8)');
    gradient.addColorStop(0.5, 'rgba(26, 26, 26, 0.6)');
    gradient.addColorStop(1, 'rgba(13, 13, 13, 0.8)');
    ctx.fillStyle = gradient;
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    
    // 绘制极光波浪
    auroraWaves.forEach((wave, index) => {
      ctx.beginPath();
      ctx.strokeStyle = wave.color;
      ctx.lineWidth = 80;
      
      for (let x = 0; x <= canvas.width; x += 5) {
        const y = wave.y + Math.sin(x * wave.frequency + time + wave.phase) * wave.amplitude;
        if (x === 0) {
          ctx.moveTo(x, y);
        } else {
          ctx.lineTo(x, y);
        }
      }
      ctx.stroke();
    });
    
    time += 0.016;
    animationFrame = requestAnimationFrame(drawAurora);
  }
  
  // 启动动画
  drawAurora();
  
  // 滚动控制极光强度
  ScrollTrigger.create({
    trigger: '.aurora-section',
    start: 'top 80%',
    end: 'bottom 20%',
    onUpdate: (self) => {
      const progress = self.progress;
      auroraWaves.forEach((wave, index) => {
        wave.amplitude = (50 + index * 20) * (1 + progress * 2);
      });
    }
  });
  
  // 清理函数
  function cleanup() {
    if (animationFrame) {
      cancelAnimationFrame(animationFrame);
    }
  }
  
  return cleanup;
}
```

class AnimationLazyLoader {
  constructor() {
    this.init();
  }
  
  init() {
    this.initIntersectionObserver();
  }
  
  initIntersectionObserver() {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const element = entry.target;
          const animationType = element.dataset.animation;
          
          switch (animationType) {
            case 'crystal-prism':

              break;

            case 'hologram':
              initScrollHologramGallery();
              break;
            case 'aurora':
              initScrollAuroraField();
              break;
          }
          
          observer.unobserve(element);
        }
      });
    }, { rootMargin: '200px', threshold: 0.1 });
    
    document.querySelectorAll('[data-animation]').forEach(el => {

    });

/**
 * 动画性能监控
 * 追踪动画性能指标
 */
class AnimationPerformanceMonitor {
  constructor() {
    this.metrics = {
      fps: 0,
      frameTime: 0,
      memory: 0
    };
    this.init();
  }
  
  init() {
    this.startFPSMonitor();
    this.startMemoryMonitor();
  }
  
  startFPSMonitor() {
    let lastTime = performance.now();
    let frames = 0;
    
    function updateFPS() {
      const now = performance.now();
      frames++;
      
      if (now - lastTime >= 1000) {
        this.metrics.fps = frames;
        frames = 0;
        lastTime = now;
        
        // 性能警告
        if (this.metrics.fps < 30) {
          console.warn('Low FPS detected:', this.metrics.fps);
          this.optimizePerformance();
        }
      }
      
      requestAnimationFrame(updateFPS);
    }
    
    requestAnimationFrame(updateFPS);
  }
  
  startMemoryMonitor() {
    if (performance.memory) {
      setInterval(() => {
        this.metrics.memory = performance.memory.usedJSHeapSize / 1048576; // MB
        
        // 内存泄漏警告
        if (this.metrics.memory > 100) {
          console.warn('High memory usage:', this.metrics.memory);
          this.optimizePerformance();
        }
      }, 5000);
    }
  }
  
  optimizePerformance() {
    // 降低动画复杂度
    document.querySelectorAll('.particle').forEach(particle => {
      if (Math.random() > 0.5) {
        particle.style.display = 'none';
      }
    });
    
    // 降低粒子数量
    const particleSystems = document.querySelectorAll('[data-particle-count]');
    particleSystems.forEach(system => {
      const count = parseInt(system.dataset.particleCount);
      system.dataset.particleCount = Math.floor(count * 0.5);
    });
  }
}
