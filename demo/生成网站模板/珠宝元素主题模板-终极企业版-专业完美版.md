## 🎯 快速开始

### 方式一: 完整部署 (推荐)

1. 复制整个模板文件结构
2. 替换图片和文本内容
3. 修改 `config.js` 中的配置参数
4. 部署到服务器

### 方式二: 模块化集成

1. 选择需要的组件 (见[可选组件清单](#🧩-可选组件清单))
2. 复制对应的 HTML/CSS/JS 代码
3. 在 `config.js` 中启用相应开关
4. 测试动画效果

## 🧩 可选组件清单

### 核心组件 (推荐必用) ⭐⭐⭐

| 组件名称 | 应用场景 | 复杂度 | 文件大小 |
|---------|---------|--------|----------|
| `ScrollCrystalPrism` | 钻石系列展示、高端定制页面 | 🟡 中 | ~8KB |
| `ScrollElegantCards` | 经典珠宝系列、品牌历史 | 🟢 低 | ~5KB |

### 增强组件 (按需选用) ⭐⭐

| 组件名称 | 应用场景 | 复杂度 | 文件大小 |
|---------|---------|--------|----------|
| `ScrollNeonPrismCards` | 夜间系列、时尚珠宝、活动专题 | 🟡 中 | ~10KB |
| `ScrollHologramGallery` | 产品展示、工艺展示、品牌历史 | 🔴 高 | ~15KB |
| `ScrollAuroraField` | Hero区域、品牌故事背景 | 🔴 高 | ~12KB |

### 移除方法

```javascript
// 在 config.js 中设置
JewelryConfig.features.enableNeonPrism = false;
JewelryConfig.features.enableHologram = false;
JewelryConfig.features.enableAurora = false;

// 在 main.js 中注释掉对应的初始化调用
// initScrollNeonPrismCards();
// initScrollHologramGallery();
// initScrollAuroraField();
```

const Utils = {
  // ==================== 安全的动画初始化 ====================
  /**
   * 安全的动画初始化包装器
   * @param {Function} initFunction - 初始化函数
   * @param {string} componentName - 组件名称
   * @returns {any} - 初始化结果
   */
  safeInit(initFunction, componentName) {
    try {
      console.log(`🚀 初始化: ${componentName}`);
      const result = initFunction();
      console.log(`✅ 成功: ${componentName}`);
      return result;
    } catch (error) {
      console.error(`❌ 失败: ${componentName}`, error);
      
      // 记录错误但不阻断
      window.animationErrors = window.animationErrors || [];
      window.animationErrors.push({
        component: componentName,
        error: error.message,
        timestamp: Date.now()
      });
      
      return null;
    }
  },

  // ==================== 防抖函数 ====================
  /**
   * 防抖函数
   * @param {Function} func - 要防抖的函数
   * @param {number} wait - 等待时间(ms)
   * @returns {Function} - 防抖后的函数
   */
  debounce(func, wait = 300) {
    let timeout;
    return function executedFunction(...args) {
      const later = () => {
        clearTimeout(timeout);
        func(...args);
      };
      clearTimeout(timeout);
      timeout = setTimeout(later, wait);
    };
  },

  // ==================== 节流函数 ====================
  /**
   * 节流函数
   * @param {Function} func - 要节流的函数
   * @param {number} limit - 时间限制(ms)
   * @returns {Function} - 节流后的函数
   */
  throttle(func, limit = 100) {
    let inThrottle;
    return function executedFunction(...args) {
      if (!inThrottle) {
        func(...args);
        inThrottle = true;
        setTimeout(() => inThrottle = false, limit);
      }
    };
  },

  // ==================== 检测元素可见性 ====================
  /**
   * 检测元素是否在视口中
   * @param {HTMLElement} element - 目标元素
   * @param {number} threshold - 可见阈值(0-1)
   * @returns {boolean} - 是否可见
   */
  isVisible(element, threshold = 0.1) {
    const rect = element.getBoundingClientRect();
    const windowHeight = window.innerHeight || document.documentElement.clientHeight;
    const windowWidth = window.innerWidth || document.documentElement.clientWidth;
    
    const visibleHeight = Math.min(
      Math.max(0, windowHeight - rect.top),
      Math.min(rect.bottom, windowHeight) - Math.max(0, rect.top)
    );
    const visibleWidth = Math.min(
      Math.max(0, windowWidth - rect.left),
      Math.min(rect.right, windowWidth) - Math.max(0, rect.left)
    );
    
    const visibleArea = visibleHeight * visibleWidth;
    const totalArea = rect.width * rect.height;
    
    return (visibleArea / totalArea) >= threshold;
  },

  // ==================== 设备类型检测 ====================
  /**
   * 检测设备类型
   * @returns {string} - 设备类型 (desktop, laptop, tablet, mobile)
   */
  getDeviceType() {
    const width = window.innerWidth;
    const touch = 'ontouchstart' in window || navigator.maxTouchPoints > 0;
    
    if (width >= 1280) return 'desktop';
    if (width >= 1024) return 'laptop';
    if (width >= 768) return 'tablet';
    if (touch) return 'mobile';
    return 'desktop';
  },

  // ==================== 性能等级检测 ====================
  /**
   * 检测设备性能等级
   * @returns {string} - 性能等级 (high, medium, low)
   */
  getPerformanceLevel() {
    const memory = navigator.deviceMemory || 4;
    const cores = navigator.hardwareConcurrency || 2;
    const connection = navigator.connection;
    
    // 低端设备
    if (memory <= 2 || cores <= 2 || (connection && connection.saveData)) {
      return 'low';
    }
    
    // 中端设备
    if (memory <= 4 || cores <= 4) {
      return 'medium';
    }
    
    // 高端设备
    return 'high';
  },

  // ==================== 平滑滚动 ====================
  /**
   * 平滑滚动到指定元素
   * @param {HTMLElement|string} target - 目标元素或选择器
   * @param {number} offset - 偏移量
   * @param {number} duration - 滚动时长
   */
  scrollTo(target, offset = 0, duration = 1000) {
    if (typeof target === 'string') {
      target = document.querySelector(target);
    }
    
    if (!target) return;
    
    const targetY = target.getBoundingClientRect().top + window.pageYOffset - offset;
    
    if (typeof gsap !== 'undefined' && gsap.to) {
      gsap.to(window, {

        ease: 'power2.inOut'
      });

        top: targetY,
        behavior: 'smooth'
      });
    }
  },

  // ==================== 随机数生成 ====================
  /**
   * 生成指定范围内的随机数
   * @param {number} min - 最小值
   * @param {number} max - 最大值
   * @returns {number} - 随机数
   */
  random(min, max) {
    return Math.random() * (max - min) + min;
  },

  // ==================== 随机整数生成 ====================
  /**
   * 生成指定范围内的随机整数
   * @param {number} min - 最小值
   * @param {number} max - 最大值
   * @returns {number} - 随机整数

  randomInt(min, max) {

class AccessibilityManager {

    ).matches;

    if (prefersReducedMotion) {
      document.body.classList.add('reduced-motion');

      if (window.JewelryConfig) {
        window.JewelryConfig.features.enableReducedMotion = true;
      }
    }

    // 监听设置变化
    window.matchMedia('(prefers-reduced-motion: reduce)').addEventListener('change', (e) => {
      if (e.matches) {
        document.body.classList.add('reduced-motion');
      } else {
        document.body.classList.remove('reduced-motion');
      }
    });
  }

  // ==================== 键盘导航 ====================
  initKeyboardNavigation() {
    // Tab键导航增强
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Tab') {
        document.body.classList.add('keyboard-navigation');
        
        // 延迟移除,保持焦点样式
        clearTimeout(this.keyboardNavTimeout);
        this.keyboardNavTimeout = setTimeout(() => {
          document.body.classList.remove('keyboard-navigation');
        }, 1000);
      }
    });

    // 跳过导航链接
    this.addSkipLinks();

    // 焦点陷阱(用于模态框)
    this.initFocusTrap();
  }

  // ==================== 添加跳过导航 ====================
  addSkipLinks() {
    const skipLink = document.createElement('a');
    skipLink.href = '#main-content';
    skipLink.className = 'skip-link';
    skipLink.textContent = '跳转到主要内容';
    skipLink.style.cssText = `
      position: absolute;
      top: -100%;
      left: 0;
      background: var(--color-gold-500);
      color: #000;
      padding: 1rem 2rem;
      z-index: 9999;
      transition: top 0.3s ease;
    `;

    skipLink.addEventListener('focus', () => {
      skipLink.style.top = '0';
    });

    skipLink.addEventListener('blur', () => {
      skipLink.style.top = '-100%';
    });

    document.body.insertBefore(skipLink, document.body.firstChild);

    // 为主要内容添加ID
    const mainContent = document.querySelector('main, [role="main"]') || document.body;
    if (!mainContent.id) {
      mainContent.id = 'main-content';
    }
  }

  // ==================== 焦点管理 ====================
  initFocusManagement() {
    // 确保所有可交互元素有焦点样式
    const interactiveElements = document.querySelectorAll(
      'a, button, input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );

    interactiveElements.forEach(el => {
      if (!el.hasAttribute('aria-label') && !el.textContent.trim()) {
        const icon = el.querySelector('svg, i');
        if (icon) {
          el.setAttribute('aria-label', icon.getAttribute('aria-label') || '交互按钮');
        }
      }
    });

    // 焦点可见性增强
    this.enhanceFocusVisibility();
  }

  // ==================== 增强焦点可见性 ====================
  enhanceFocusVisibility() {
    const style = document.createElement('style');
    style.textContent = `
      .keyboard-navigation *:focus {
        outline: 3px solid var(--color-gold-500) !important;
        outline-offset: 2px !important;
      }
      
      .keyboard-navigation *:focus:not(:focus-visible) {
        outline: 3px solid var(--color-gold-500) !important;
      }
      
      *:focus-visible {
        outline: 3px solid var(--color-gold-500) !important;
        outline-offset: 2px !important;
      }
    `;
    document.head.appendChild(style);
  }

  // ==================== 焦点陷阱 ====================
  initFocusTrap() {
    // 用于模态框和对话框的焦点管理
    this.trapFocus = function(container) {
      const focusableElements = container.querySelectorAll(
        'a[href], button:not([disabled]), input:not([disabled]), select:not([disabled]), textarea:not([disabled]), [tabindex]:not([tabindex="-1"])'
      );
      
      const firstElement = focusableElements[0];
      const lastElement = focusableElements[focusableElements.length - 1];

      container.addEventListener('keydown', (e) => {
        if (e.key === 'Tab') {
          if (e.shiftKey && document.activeElement === firstElement) {
            e.preventDefault();
            lastElement.focus();
          } else if (!e.shiftKey && document.activeElement === lastElement) {
            e.preventDefault();
            firstElement.focus();
          }
        }
      });
    };
  }

  // ==================== 屏幕阅读器支持 ====================
  initScreenReaderSupport() {
    // 添加ARIA角色
    this.addARIARoles();

    // 添加实时区域

    // 隐藏装饰性内容

  }

    // 导航区域
    const nav = document.querySelector('nav');
    if (nav && !nav.hasAttribute('role')) {
      nav.setAttribute('role', 'navigation');
    }

    // 主要内容
    const main = document.querySelector('main');
    if (main && !main.hasAttribute('role')) {
      main.setAttribute('role', 'main');
    }

    // 页脚
    const footer = document.querySelector('footer');
    if (footer && !footer.hasAttribute('role')) {
      footer.setAttribute('role', 'contentinfo');
    }

    // 卡片组

    cardGroups.forEach(group => {

        group.setAttribute('role', 'list');

        cards.forEach(card => card.setAttribute('role', 'listitem'));

    });

      margin: -1px;

      liveRegion.textContent = message;
    };

    if (prefersLargeText) {
      document.documentElement.classList.add('large-text');
    }
  }
}

/**
 * ============================================
 * 💎 珠宝企业官网 - 性能优化系统
 * ============================================
 * 版本: 3.0.0
 * ============================================
 */

class PerformanceManager {
  constructor() {
    this.metrics = {
      fps: 60,
      frameTime: 0,
      memory: 0,
      loadTime: 0
    };
    
    this.optimizationLevel = 'none'; // none, low, medium, high
    
    this.init();
  }

  init() {
    this.measureLoadTime();
    this.startFPSMonitor();
    this.startMemoryMonitor();
    this.initPerformanceObserver();
    this.checkInitialPerformance();
  }

  // ==================== 测量页面加载时间 ====================
  measureLoadTime() {
    if (window.performance) {
      window.addEventListener('load', () => {
        const navigation = performance.getEntriesByType('navigation')[0];
        this.metrics.loadTime = navigation.loadEventEnd - navigation.fetchStart;
        
        console.log(`📊 页面加载时间: ${this.metrics.loadTime.toFixed(2)}ms`);
        
        // 发送到分析工具
        this.reportToAnalytics('page_load', { loadTime: this.metrics.loadTime });
      });
    }
  }

  // ==================== FPS监控 ====================
  startFPSMonitor() {
    let lastTime = performance.now();
    let frames = 0;
    let lastReportTime = lastTime;

    const updateFPS = () => {
      const now = performance.now();
      frames++;

      // 每秒计算一次FPS
      if (now - lastReportTime >= 1000) {
        this.metrics.fps = frames;
        frames = 0;
        lastReportTime = now;

        // 性能警告
        if (this.metrics.fps < 30) {
          console.warn(`⚠️ 低FPS检测: ${this.metrics.fps}fps`);
          this.optimizePerformance('low');
        } else if (this.metrics.fps < 45) {
          console.warn(`⚠️ FPS偏低: ${this.metrics.fps}fps`);
          this.optimizePerformance('medium');
        }
      }

      requestAnimationFrame(updateFPS);
    };

    updateFPS();
  }

  // ==================== 内存监控 ====================
  startMemoryMonitor() {
    if (!performance.memory) {
      console.warn('⚠️ 内存API不支持');
      return;
    }

    setInterval(() => {
      this.metrics.memory = performance.memory.usedJSHeapSize / 1048576; // MB

      // 内存泄漏警告
      if (this.metrics.memory > 100) {
        console.warn(`⚠️ 高内存使用: ${this.metrics.memory.toFixed(2)}MB`);
        this.optimizePerformance('high');
      }

      // 发送到分析工具
      this.reportToAnalytics('memory_usage', { memory: this.metrics.memory });
    }, 5000);
  }

  // ==================== 性能观察器 ====================
  initPerformanceObserver() {
    if (!window.PerformanceObserver) return;

    // 长任务监控
    const longTaskObserver = new PerformanceObserver((list) => {
      list.getEntries().forEach(entry => {
        console.warn(`⚠️ 长任务检测: ${entry.duration.toFixed(2)}ms`);
        this.reportToAnalytics('long_task', { duration: entry.duration });
      });
    });
    
    longTaskObserver.observe({ entryTypes: ['longtask'] });

    // 布局偏移监控
    const layoutShiftObserver = new PerformanceObserver((list) => {
      list.getEntries().forEach(entry => {
        if (!entry.hadRecentInput) {
          console.warn(`⚠️ 布局偏移: ${entry.value.toFixed(4)}`);
        }
      });
    });
    
    layoutShiftObserver.observe({ entryTypes: ['layout-shift'] });
  }

  // ==================== 检查初始性能 ====================
  checkInitialPerformance() {
    const performanceLevel = Utils.getPerformanceLevel();
    console.log(`📊 设备性能等级: ${performanceLevel}`);

    // 根据性能等级预优化
    if (performanceLevel === 'low') {
      this.optimizePerformance('high');
    } else if (performanceLevel === 'medium') {
      this.optimizePerformance('medium');
    }
  }

  // ==================== 性能优化 ====================
  /**
   * 根据优化级别调整动画质量
   * @param {string} level - 优化级别 (low, medium, high)
   */
  optimizePerformance(level) {
    if (this.optimizationLevel === level) return;
    this.optimizationLevel = level;

    console.log(`🔧 性能优化级别: ${level}`);

    switch (level) {
      case 'low':
        // 轻度优化
        this.disableNonEssentialAnimations();
        break;
        
      case 'medium':
        // 中度优化
        this.reduceParticleCount(0.5);
        this.disableNonEssentialAnimations();
        this.reduceAnimationQuality('medium');
        break;
        
      case 'high':
        // 高度优化
        this.reduceParticleCount(0.2);
        this.disableNonEssentialAnimations();
        this.reduceAnimationQuality('low');
        this.disableComplexEffects();
        break;
    }

    // 通知用户
    if (window.accessibility && window.accessibility.announce) {
      window.accessibility.announce('已启用性能优化模式');
    }
  }

  // ==================== 禁用非必要动画 ====================
  disableNonEssentialAnimations() {
    // 禁用粒子系统
    if (window.JewelryConfig) {
      window.JewelryConfig.features.enableParticles = false;
    }

    // 隐藏粒子元素
    const particles = document.querySelectorAll('.particle');
    particles.forEach(p => p.style.display = 'none');
  }

  // ==================== 减少粒子数量 ====================
  /**
   * @param {number} ratio - 剩余比例 (0-1)
   */
  reduceParticleCount(ratio) {
    const particleSystems = document.querySelectorAll('[data-particle-count]');
    particleSystems.forEach(system => {
      const originalCount = parseInt(system.dataset.particleCount);
      const newCount = Math.floor(originalCount * ratio);
      system.dataset.particleCount = newCount;

      // 隐藏多余粒子
      const particles = system.querySelectorAll('.particle');
      particles.forEach((particle, index) => {
        if (index >= newCount) {
          particle.style.display = 'none';
        }
      });
    });
  }

  // ==================== 降低动画质量 ====================
  /**
   * @param {string} quality - 质量级别 (low, medium, high)
   */
  reduceAnimationQuality(quality) {
    const qualityMap = {
      low: { scale: 0.5, duration: 0.3 },
      medium: { scale: 1.0, duration: 0.6 },
      high: { scale: 1.5, duration: 1.0 }
    };

    const config = qualityMap[quality] || qualityMap.high;

    // 缩短动画时长
    const animations = document.getAnimations();
    animations.forEach(anim => {
      anim.updatePlaybackRate(config.duration);
    });

    // 降低Canvas质量
    const canvases = document.querySelectorAll('canvas');
    canvases.forEach(canvas => {
      const ctx = canvas.getContext('2d');
      if (ctx) {
        ctx.imageSmoothingEnabled = quality === 'low';
      }
    });
  }

  // ==================== 禁用复杂效果 ====================
  disableComplexEffects() {
    // 禁用backdrop-filter
    const glassElements = document.querySelectorAll('.glass, .glass-card');
    glassElements.forEach(el => {
      el.style.backdropFilter = 'none';
      el.style.webkitBackdropFilter = 'none';
    });

    // 禁用3D变换
    const elements3D = document.querySelectorAll('[style*="transform-style: preserve-3d"]');
    elements3D.forEach(el => {
      el.style.transformStyle = 'flat';
    });

    // 禁用阴影
    const elementsWithShadow = document.querySelectorAll('[style*="box-shadow"]');
    elementsWithShadow.forEach(el => {
      el.style.boxShadow = 'none';
    });
  }

  // ==================== 懒加载动画 ====================
  /**
   * 创建懒加载观察器
   * @returns {IntersectionObserver}
   */
  createLazyLoader() {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const element = entry.target;
          const animationType = element.dataset.animation;

          // 延迟初始化,避免阻塞渲染
          requestIdleCallback(() => {
            this.initAnimation(animationType, element);
          });

          observer.unobserve(element);
        }
      });
    }, {
      rootMargin: '200px',
      threshold: 0.1
    });

    return observer;
  }

  // ==================== 初始化懒加载动画 ====================
  initAnimation(type, element) {
    console.log(`🚀 懒加载初始化: ${type}`);

    switch (type) {
      case 'crystal-prism':
        if (typeof initScrollCrystalPrism === 'function') {
          initScrollCrystalPrism();
        }
        break;
      case 'neon-prism':
        if (typeof initScrollNeonPrismCards === 'function') {
          initScrollNeonPrismCards();
        }
        break;
      case 'hologram':
        if (typeof initScrollHologramGallery === 'function') {
          initScrollHologramGallery();
        }
        break;
      case 'aurora':
        if (typeof initScrollAuroraField === 'function') {
          initScrollAuroraField();
        }
        break;
    }
  }

  // ==================== 报告到分析工具 ====================
  reportToAnalytics(event, data) {
    // 这里可以集成Google Analytics或其他分析工具
    if (typeof gtag === 'function') {
      gtag('event', event, data);
    }
  }

  // ==================== 获取性能报告 ====================
  getReport() {
    return {
      fps: this.metrics.fps,
      frameTime: this.metrics.frameTime,
      memory: this.metrics.memory.toFixed(2),
      loadTime: this.metrics.loadTime.toFixed(2),
      optimizationLevel: this.optimizationLevel,
      deviceType: Utils.getDeviceType(),
      performanceLevel: Utils.getPerformanceLevel()
    };
  }
}

class CompatibilityChecker {
  constructor() {
    this.features = {};
    this.missingFeatures = [];
    this.init();
  }

  init() {
    this.checkGSAP();
    this.checkCanvas();
    this.checkWebGL();
    this.checkCSSFeatures();

  // ==================== 检查GSAP ====================
  checkGSAP() {
    this.features.gsap = typeof gsap !== 'undefined';
    this.features.scrollTrigger = typeof ScrollTrigger !== 'undefined';
    this.features.scrollToPlugin = typeof ScrollToPlugin !== 'undefined';

    if (!this.features.gsap) {
      this.missingFeatures.push('GSAP');
    }
    if (!this.features.scrollTrigger) {
      this.missingFeatures.push('ScrollTrigger');
    }
  }

  // ==================== 检查Canvas ====================
  checkCanvas() {
    const canvas = document.createElement('canvas');
    this.features.canvas = !!canvas.getContext;

    if (!this.features.canvas) {
      this.missingFeatures.push('Canvas API');
    }
  }

  // ==================== 检查WebGL ====================
  checkWebGL() {
    try {
      const canvas = document.createElement('canvas');
      this.features.webgl = !!window.WebGLRenderingContext && 
                            !!canvas.getContext('webgl');
    } catch (e) {
      this.features.webgl = false;
    }

    if (!this.features.webgl) {
      console.warn('⚠️ WebGL不支持,部分3D效果将降级');
    }
  }

  // ==================== 检查CSS特性 ====================
  checkCSSFeatures() {
    this.features.backdropFilter = CSS.supports('backdrop-filter', 'blur(10px)');
    this.features.mixBlendMode = CSS.supports('mix-blend-mode', 'multiply');
    this.features.clipPath = CSS.supports('clip-path', 'circle(50%)');
    this.features.transform3D = CSS.supports('transform', 'translate3d(0,0,0)');
    this.features.willChange = CSS.supports('will-change', 'transform');
    this.features.cssGrid = CSS.supports('display', 'grid');
    this.features.flexbox = CSS.supports('display', 'flex');

    if (!this.features.backdropFilter) {
      this.missingFeatures.push('backdrop-filter');
    }
    if (!this.features.mixBlendMode) {
      this.missingFeatures.push('mix-blend-mode');
    }
  }

  // ==================== 检查ES6特性 ====================
  checkES6() {
    try {
      // 箭头函数
      const arrow = () => {};
      
      // 模板字符串
      const template = `test ${1 + 1}`;
      
      // 解构赋值
      const { a } = { a: 1 };
      
      // 展开运算符
      const spread = [...[1, 2, 3]];
      
      // Promise
      const promise = new Promise(() => {});
      
      // Class
      class Test {}
      
      this.features.es6 = true;
    } catch (e) {
      this.features.es6 = false;
      this.missingFeatures.push('ES6');
    }
  }

  // ==================== 检查Web APIs ====================
  checkAPIs() {
    this.features.intersectionObserver = 'IntersectionObserver' in window;
    this.features.requestAnimationFrame = 'requestAnimationFrame' in window;
    this.performance = 'performance' in window;
    this.localStorage = 'localStorage' in window;
    this.fetch = 'fetch' in window;

    if (!this.features.intersectionObserver) {
      console.warn('⚠️ IntersectionObserver不支持,将使用polyfill');
    }
  }

  // ==================== 生成兼容性报告 ====================
  generateReport() {
    console.group('🔍 浏览器兼容性检测');
    
    console.table(this.features);
    
    if (this.missingFeatures.length > 0) {
      console.warn('⚠️ 以下特性不被支持:', this.missingFeatures.join(', '));
    } else {
      console.log('✅ 所有特性均支持');
    }

    console.groupEnd();

    return this.getReport();
  }

  // ==================== 获取兼容性报告 ====================
  getReport() {
    return {
      userAgent: navigator.userAgent,
      features: this.features,
      missingFeatures: this.missingFeatures,
      isCompatible: this.missingFeatures.length === 0,
      browser: this.detectBrowser(),
      os: this.detectOS()
    };
  }

  // ==================== 检测浏览器 ====================
  detectBrowser() {
    const ua = navigator.userAgent;
    
    if (ua.indexOf('Chrome') > -1) return 'Chrome';
    if (ua.indexOf('Firefox') > -1) return 'Firefox';
    if (ua.indexOf('Safari') > -1) return 'Safari';
    if (ua.indexOf('Edge') > -1) return 'Edge';
    if (ua.indexOf('Opera') > -1) return 'Opera';
    if (ua.indexOf('MSIE') > -1 || ua.indexOf('Trident') > -1) return 'IE';
    
    return 'Unknown';
  }

  // ==================== 检测操作系统 ====================
  detectOS() {
    const ua = navigator.userAgent;
    
    if (ua.indexOf('Win') > -1) return 'Windows';
    if (ua.indexOf('Mac') > -1) return 'macOS';
    if (ua.indexOf('Linux') > -1) return 'Linux';
    if (ua.indexOf('Android') > -1) return 'Android';
    if (ua.indexOf('iOS') > -1 || ua.indexOf('iPhone') > -1 || ua.indexOf('iPad') > -1) return 'iOS';
    
    return 'Unknown';
  }

  // ==================== 应用降级方案 ====================
  applyFallbacks() {
    // 不支持backdrop-filter
    if (!this.features.backdropFilter) {
      document.body.classList.add('no-backdrop-filter');

    }

      document.body.classList.add('no-3d');
    }

    // 不支持的特性太多,启用降级模式
    if (this.missingFeatures.length >= 3) {
      document.body.classList.add('fallback-mode');
      console.log('⚠️ 已启用降级模式');
    }
  }
}

(function() {
  'use strict';

  // ==================== 全局状态 ====================
  const App = {
    initialized: false,
    components: {},
    cleanupFunctions: []
  };

  // ==================== 错误处理 ====================
  function handleError(error, context) {
    console.error(`❌ 错误 [${context}]:`, error);

    if (typeof Lenis === 'undefined') return;

    const lenis = new Lenis({
      duration: 1.2,
      easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
      smooth: true
    });

    function raf(time) {
      lenis.raf(time);
      requestAnimationFrame(raf);
    }

    requestAnimationFrame(raf);

    // 集成GSAP ScrollTrigger
    if (typeof ScrollTrigger !== 'undefined') {

      ScrollTrigger.scrollerProxy(lenis.root, {
        scrollTop(value) {

            ? lenis.scrollTo(value, { immediate: true })
            : lenis.scroll;

        getBoundingClientRect() {
          return {
            top: 0,
            left: 0,
            width: window.innerWidth,
            height: window.innerHeight

      gsap.ticker.add((time) => lenis.raf(time * 1000));

    initApp();
  }

  // ==================== 导出 ====================
  if (typeof module !== 'undefined' && module.exports) {
    module.exports = { App, initApp };
  }

})();

## 📞 技术支持

### 常见问题

**Q: 动画卡顿怎么办?**  
A: 检查性能监控报告,调用 `JewelryApp.optimizePerformance('high')` 进行优化。

**Q: 浏览器不支持怎么办?**  
A: 查看兼容性检测报告 `JewelryApp.getCompatibility()`,系统会自动降级。

**Q: 如何禁用某些动画?**  
A: 在 `config.js` 中修改 `JewelryConfig.features` 对应的开关。

**Q: 如何自定义颜色?**  
A: 在 `config.js` 中修改 `JewelryConfig.colors` 对象。
