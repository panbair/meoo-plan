    const speed = parseInt(element.dataset.speed) || 100;
    
    element.textContent = '';
    
    // 创建光标
    const cursor = document.createElement('span');
    cursor.className = 'typewriter-cursor';
    cursor.textContent = '|';
    cursor.style.cssText = `
      display: inline-block;
      animation: blink 1s step-end infinite;
      color: #D4AF37;
      font-weight: bold;
    `;
    
    // 添加光标样式
    if (!document.querySelector('#typewriter-style')) {
      const style = document.createElement('style');
      style.id = 'typewriter-style';
      style.textContent = `
        @keyframes blink {
          0%, 100% { opacity: 1; }
          50% { opacity: 0; }
        }
      `;
      document.head.appendChild(style);
    }
    
    let index = 0;
    
    // 触发动画
    ScrollTrigger.create({
      trigger: element,
      start: 'top 80%',
      onEnter: () => {
        typeText();
      }
    });
    
    function typeText() {
      element.appendChild(cursor);
      
      const typeInterval = setInterval(() => {
        if (index < text.length) {
          const char = document.createTextNode(text[index]);
          element.insertBefore(char, cursor);
          index++;
        } else {
          clearInterval(typeInterval);
        }
      }, speed);
    }
  });
}

/**
 * 智能导航系统 - 增强版
 * 包含：滚动检测、平滑滚动、智能预测、自适应布局、语音导航、精美动画

class SmartNavigation {
  constructor() {
    this.navbar = document.querySelector('.navbar');
    this.navLinks = document.querySelectorAll('.nav-link');
    this.sections = document.querySelectorAll('section');
    this.scrollHistory = [];
    this.scrollSpeed = 0;

    this.initScrollDetection();

  initScrollDetection() {
    let ticking = false;
    
    window.addEventListener('scroll', () => {
      if (!ticking) {
        window.requestAnimationFrame(() => {
          this.handleScroll();
          ticking = false;
        });
        ticking = true;
      }
    });
  }
  
  handleScroll() {
    const scrollY = window.scrollY;
    
    // 计算滚动速度
    this.scrollSpeed = Math.abs(scrollY - this.lastScrollY);
    this.lastScrollY = scrollY;
    
    // 更新导航栏状态
    this.updateNavbarState(scrollY);
    
    // 更新当前区域
    this.updateActiveSection(scrollY);
    
    // 记录滚动历史
    this.recordScrollHistory(scrollY);
    
    // Logo 动态效果

    if (scrollY > 50) {

      // 快速滚动时压缩
      if (this.scrollSpeed > 50) {
        this.navbar.classList.add('compressed');
      } else {
        this.navbar.classList.remove('compressed');

      });

    if (!logo) return;

    const glowIntensity = 5 + scrollProgress * 20;

    const ripple = document.createElement('span');
    ripple.className = 'ripple';
    
    const rect = element.getBoundingClientRect();
    const size = Math.max(rect.width, rect.height);
    
    ripple.style.width = ripple.style.height = `${size}px`;
    ripple.style.left = `${event.clientX - rect.left - size / 2}px`;
    ripple.style.top = `${event.clientY - rect.top - size / 2}px`;
    
    element.appendChild(ripple);
    
    gsap.fromTo(ripple,
      { scale: 0, opacity: 0.5 },
      {
        scale: 2,
        opacity: 0,
        duration: 0.6,
        ease: 'power2.out',
        onComplete: () => ripple.remove()
      }
    );
  }
  
  // 激活当前链接

    this.updateActiveLink();
    
    window.addEventListener('scroll', () => {

    });
  }

    this.sections.forEach(section => {
      const sectionTop = section.offsetTop;
      const sectionHeight = section.clientHeight;

      position: scrollY,
      timestamp: Date.now(),
      section: this.currentSection
    });

    if (this.scrollHistory.length > 100) {
      this.scrollHistory.shift();
    }
  }
  
  // 销毁
  destroy() {
    if (this.logoGlowTimeline) {
      this.logoGlowTimeline.kill();
    }
  }

/**
 * 高级粒子系统 - 增强版
 * 包含：钻石闪烁粒子、金粉粒子、连接线粒子、流星效果、星尘爆炸
 */
class ParticleSystem {
  constructor(container, options = {}) {
    this.container = container;
    this.particles = [];
    this.lines = [];
    this.meteors = [];
    this.explosions = [];
    this.animationId = null;
    
    this.config = {
      count: options.count || 80,
      diamondRatio: options.diamondRatio || 0.2,
      goldRatio: options.goldRatio || 0.5,
      starRatio: options.starRatio || 0.3,
      lineEnabled: options.lineEnabled || true,
      lineDistance: options.lineDistance || 150,
      meteorEnabled: options.meteorEnabled || true,
      meteorInterval: options.meteorInterval || 3000,
      mouseEnabled: options.mouseEnabled || true
    };
    
    this.mouse = { x: -1, y: -1 };
    
    this.init();
  }
  
  init() {
    this.createParticles();
    this.createLines();
    this.initMouseInteraction();
    this.initMeteors();
    this.animate();
  }
  
  createParticles() {
    for (let i = 0; i < this.config.count; i++) {
      const type = this.getParticleType(i);
      const particle = this.createParticle(type);
      this.particles.push(particle);
      this.container.appendChild(particle.element);
    }
  }
  
  getParticleType(index) {
    const random = Math.random();
    
    if (random < this.config.diamondRatio) {
      return 'diamond';
    } else if (random < this.config.diamondRatio + this.config.goldRatio) {
      return 'gold';
    } else if (random < this.config.diamondRatio + this.config.goldRatio + this.config.starRatio) {
      return 'star';
    } else {
      return 'basic';
    }
  }
  
  createParticle(type) {
    const element = document.createElement('div');
    element.className = 'particle';
    
    // 随机位置
    const x = Math.random() * 100;
    const y = Math.random() * 100;
    
    // 根据类型设置大小
    let size;
    switch (type) {
      case 'diamond':
        size = Math.random() * 5 + 3;
        break;
      case 'star':
        size = Math.random() * 4 + 2;
        break;
      case 'gold':
        size = Math.random() * 3 + 1;
        break;
      default:
        size = Math.random() * 2 + 0.5;
    }
    
    // 随机透明度
    const opacity = Math.random() * 0.7 + 0.3;
    
    element.style.left = `${x}%`;
    element.style.top = `${y}%`;
    element.style.width = `${size}px`;
    element.style.height = `${size}px`;
    element.style.opacity = opacity;
    
    // 根据类型设置样式
    if (type === 'diamond') {
      element.classList.add('particle-diamond');
      element.style.boxShadow = '0 0 15px #D4AF37, 0 0 30px #F5DEB3';
      element.style.background = 'radial-gradient(circle, #FFF, #D4AF37)';
    } else if (type === 'gold') {
      element.classList.add('particle-gold');
      element.style.background = 'linear-gradient(135deg, #D4AF37, #F5DEB3)';
      element.style.borderRadius = '50%';
    } else if (type === 'star') {
      element.classList.add('particle-star');
      element.innerHTML = '✦';
      element.style.fontSize = `${size}px`;
      element.style.color = '#D4AF37';
      element.style.textShadow = '0 0 10px rgba(212, 175, 55, 0.8)';
    } else {
      element.classList.add('particle-basic');
      element.style.background = 'rgba(255, 255, 255, 0.3)';
      element.style.borderRadius = '50%';
    }
    
    // 动画参数
    const particle = {
      element,
      type,
      x,
      y,
      size,
      opacity,
      vx: (Math.random() - 0.5) * 0.3,
      vy: (Math.random() - 0.5) * 0.3 - 0.2,
      phase: Math.random() * Math.PI * 2,
      speed: Math.random() * 2 + 3,
      amplitude: Math.random() * 60 + 20,
      twinkleSpeed: Math.random() * 3 + 2,
      rotateSpeed: (Math.random() - 0.5) * 0.5
    };
    
    return particle;
  }
  
  createLines() {
    if (!this.config.lineEnabled) return;
    
    const canvas = document.createElement('canvas');
    canvas.className = 'particle-lines';
    canvas.style.cssText = `
      position: absolute;
      top: 0;
      left: 0;
      width: 100%;
      height: 100%;
      pointer-events: none;
      z-index: 0;
    `;
    
    this.container.insertBefore(canvas, this.container.firstChild);
    this.linesCanvas = canvas;
    this.linesCtx = canvas.getContext('2d');
    
    this.resizeLines();
    window.addEventListener('resize', () => this.resizeLines());
  }
  
  resizeLines() {
    this.linesCanvas.width = this.container.offsetWidth;
    this.linesCanvas.height = this.container.offsetHeight;
  }
  
  initMouseInteraction() {
    if (!this.config.mouseEnabled) return;
    
    this.container.addEventListener('mousemove', (e) => {
      const rect = this.container.getBoundingClientRect();
      this.mouse.x = e.clientX - rect.left;
      this.mouse.y = e.clientY - rect.top;
      
      // 鼠标附近的粒子被吸引
      this.particles.forEach(particle => {
        const px = (particle.x / 100) * this.linesCanvas.width;
        const py = (particle.y / 100) * this.linesCanvas.height;
        
        const dx = this.mouse.x - px;
        const dy = this.mouse.y - py;
        const distance = Math.sqrt(dx * dx + dy * dy);
        
        if (distance < 100) {
          const force = (100 - distance) / 100;
          particle.vx += dx / distance * force * 0.02;
          particle.vy += dy / distance * force * 0.02;
        }
      });
    });
    
    this.container.addEventListener('click', (e) => {
      this.createExplosion(e.offsetX, e.offsetY);
    });
  }
  
  initMeteors() {
    if (!this.config.meteorEnabled) return;
    
    setInterval(() => {
      this.createMeteor();
    }, this.config.meteorInterval);
  }
  
  createMeteor() {
    const meteor = document.createElement('div');
    meteor.className = 'meteor';
    
    const startX = Math.random() * this.container.offsetWidth;
    const startY = -50;
    const angle = Math.PI / 4 + Math.random() * 0.3;
    
    meteor.style.left = `${startX}px`;
    meteor.style.top = `${startY}px`;
    meteor.style.width = '100px';
    meteor.style.height = '2px';
    meteor.style.background = 'linear-gradient(90deg, transparent, #D4AF37, #FFF)';
    meteor.style.transform = `rotate(${angle}rad)`;
    meteor.style.boxShadow = '0 0 10px #D4AF37';
    
    this.container.appendChild(meteor);
    
    const meteorObj = {
      element: meteor,
      x: startX,
      y: startY,
      vx: Math.cos(angle) * 8,
      vy: Math.sin(angle) * 8,
      life: 1
    };
    
    this.meteors.push(meteorObj);
  }
  
  createExplosion(x, y) {
    const particleCount = 15;
    const explosion = {
      x: x / this.container.offsetWidth * 100,
      y: y / this.container.offsetHeight * 100,
      particles: []
    };
    
    for (let i = 0; i < particleCount; i++) {
      const angle = (Math.PI * 2 / particleCount) * i;
      const speed = Math.random() * 3 + 2;
      
      const particle = document.createElement('div');
      particle.className = 'explosion-particle';
      particle.style.cssText = `
        position: absolute;
        left: ${x}px;
        top: ${y}px;
        width: 4px;
        height: 4px;
        background: #D4AF37;
        border-radius: 50%;
        box-shadow: 0 0 10px #D4AF37;
        pointer-events: none;
      `;
      
      this.container.appendChild(particle);
      
      explosion.particles.push({
        element: particle,
        x: x,
        y: y,
        vx: Math.cos(angle) * speed,
        vy: Math.sin(angle) * speed,
        life: 1,
        decay: Math.random() * 0.03 + 0.02
      });
    }
    
    this.explosions.push(explosion);
  }
  
  animate() {
    const animateFrame = () => {
      this.updateParticles();
      this.updateMeteors();
      this.updateExplosions();
      
      if (this.config.lineEnabled) {
        this.drawLines();
      }
      
      this.animationId = requestAnimationFrame(animateFrame);
    };
    
    this.animationId = requestAnimationFrame(animateFrame);
  }
  
  updateParticles() {
    this.particles.forEach(particle => {
      // 更新相位
      particle.phase += 0.02 * particle.twinkleSpeed;
      
      // 水平移动
      particle.x += particle.vx;
      
      // 垂直移动（波浪 + 鼠标交互）
      const wave = Math.sin(particle.phase) * 0.08;
      particle.y += particle.vy + wave;
      
      // 旋转
      if (particle.element.style.transform) {
        const currentRotation = parseFloat(particle.element.dataset.rotation || 0);
        particle.element.dataset.rotation = currentRotation + particle.rotateSpeed;
        particle.element.style.transform = `rotate(${currentRotation + particle.rotateSpeed}rad)`;
      }
      
      // 边界检查
      if (particle.x < 0) particle.x = 100;
      if (particle.x > 100) particle.x = 0;
      if (particle.y < 0) particle.y = 100;
      if (particle.y > 100) particle.y = 0;
      
      // 更新 DOM
      particle.element.style.left = `${particle.x}%`;
      particle.element.style.top = `${particle.y}%`;
      
      // 钻石和星星闪烁
      if (particle.type === 'diamond' || particle.type === 'star') {
        const twinkle = Math.sin(particle.phase * 2) * 0.4 + 0.6;
        particle.element.style.opacity = particle.opacity * twinkle;
        
        // 钻石额外光效
        if (particle.type === 'diamond') {
          const glow = Math.sin(particle.phase) * 10 + 15;
          particle.element.style.boxShadow = `0 0 ${glow}px #D4AF37, 0 0 ${glow * 2}px #F5DEB3`;
        }
      }
      
      // 金色粒子大小脉动
      if (particle.type === 'gold') {
        const pulse = Math.sin(particle.phase) * 0.3 + 1;
        const newSize = particle.size * pulse;
        particle.element.style.width = `${newSize}px`;
        particle.element.style.height = `${newSize}px`;
      }
    });
  }
  
  updateMeteors() {
    for (let i = this.meteors.length - 1; i >= 0; i--) {
      const meteor = this.meteors[i];
      
      meteor.x += meteor.vx;
      meteor.y += meteor.vy;
      meteor.life -= 0.02;
      
      meteor.element.style.left = `${meteor.x}px`;
      meteor.element.style.top = `${meteor.y}px`;
      meteor.element.style.opacity = meteor.life;
      
      if (meteor.life <= 0 || meteor.y > this.container.offsetHeight + 100) {
        meteor.element.remove();
        this.meteors.splice(i, 1);
      }
    }
  }
  
  updateExplosions() {
    for (let i = this.explosions.length - 1; i >= 0; i--) {
      const explosion = this.explosions[i];
      let alive = false;
      
      explosion.particles.forEach((particle, index) => {
        if (particle.life > 0) {
          alive = true;
          particle.x += particle.vx;
          particle.y += particle.vy;
          particle.life -= particle.decay;
          
          particle.element.style.left = `${particle.x}px`;
          particle.element.style.top = `${particle.y}px`;
          particle.element.style.opacity = particle.life;
          particle.element.style.transform = `scale(${particle.life})`;
        } else {
          particle.element.remove();
          explosion.particles.splice(index, 1);
        }
      });
      
      if (!alive) {
        this.explosions.splice(i, 1);
      }
    }
  }
  
  drawLines() {
    const ctx = this.linesCtx;
    const width = this.linesCanvas.width;
    const height = this.linesCanvas.height;
    
    ctx.clearRect(0, 0, width, height);
    
    // 绘制连接线
    this.particles.forEach((p1, i) => {
      this.particles.forEach((p2, j) => {
        if (i >= j) return;
        
        const x1 = (p1.x / 100) * width;
        const y1 = (p1.y / 100) * height;
        const x2 = (p2.x / 100) * width;
        const y2 = (p2.y / 100) * height;
        
        const distance = Math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2);
        
        if (distance < this.config.lineDistance) {
          const opacity = (1 - distance / this.config.lineDistance) * 0.4;
          
          ctx.beginPath();
          ctx.moveTo(x1, y1);
          ctx.lineTo(x2, y2);
          ctx.strokeStyle = `rgba(212, 175, 55, ${opacity})`;
          ctx.lineWidth = 1;
          ctx.stroke();
          
          // 端点光晕
          ctx.beginPath();
          ctx.arc(x1, y1, 2, 0, Math.PI * 2);
          ctx.arc(x2, y2, 2, 0, Math.PI * 2);
          ctx.fillStyle = `rgba(212, 175, 55, ${opacity * 0.5})`;
          ctx.fill();
        }
      });
    });
    
    // 鼠标附近连接线
    if (this.mouse.x >= 0) {
      this.particles.forEach(particle => {
        const px = (particle.x / 100) * width;
        const py = (particle.y / 100) * height;
        
        const dx = this.mouse.x - px;
        const dy = this.mouse.y - py;
        const distance = Math.sqrt(dx * dx + dy * dy);
        
        if (distance < 150) {
          const opacity = (1 - distance / 150) * 0.6;
          
          ctx.beginPath();
          ctx.moveTo(this.mouse.x, this.mouse.y);
          ctx.lineTo(px, py);
          ctx.strokeStyle = `rgba(255, 215, 0, ${opacity})`;
          ctx.lineWidth = 2;
          ctx.stroke();
        }
      });
    }
  }
  
  destroy() {
    if (this.animationId) {
      cancelAnimationFrame(this.animationId);
    }
    
    this.particles.forEach(particle => {
      particle.element.remove();
    });
    
    this.meteors.forEach(meteor => {
      meteor.element.remove();
    });
    
    this.explosions.forEach(explosion => {
      explosion.particles.forEach(particle => {
        particle.element.remove();
      });
    });
    
    this.particles = [];
    this.meteors = [];
    this.explosions = [];
  }
}

function initBackToTop() {

  const progressRing = backToTop?.querySelector('.progress-ring-circle');
  const progressBg = backToTop?.querySelector('.progress-ring-bg');

  function updateBackToTop() {

    const windowHeight = window.innerHeight;
    const documentHeight = document.documentElement.scrollHeight;
    
    // 显示/隐藏
    if (scrollY > windowHeight * 0.4) {
      backToTop.classList.add('visible');

        gsap.fromTo(backToTop,
          {
            y: 50,
            opacity: 0,
            scale: 0.5
          },
          {
            y: 0,
            opacity: 1,
            scale: 1,
            duration: 0.6,
            ease: 'back.out(1.7)'
          }
        );
        backToTop.dataset.animated = 'true';

      backToTop.classList.remove('visible');
      backToTop.dataset.animated = 'false';
    }
    
    // 更新进度
    if (progressRing) {
      const scrollPercent = scrollY / (documentHeight - windowHeight);
      const offset = circumference - (scrollPercent * circumference);
      progressRing.style.strokeDashoffset = offset;
      
      // 进度圈颜色变化
      const hue = 45 + scrollPercent * 15; // 金色到橙金色
      progressRing.style.stroke = `hsl(${hue}, 80%, 50%)`;
    }
    
    // 旋转按钮（滚动时）
    const rotation = scrollY * 0.5;
    backToTop.style.transform = `rotate(${rotation}deg)`;
  }
  
  // 悬停效果
  backToTop.addEventListener('mouseenter', () => {
    // 放大
    gsap.to(backToTop, {
      scale: 1.1,
      duration: 0.3,
      ease: 'back.out(2)'
    });
    
    // 光晕增强
    gsap.to(glow, {
      scale: 1.5,
      opacity: 0.8,
      duration: 0.4,
      ease: 'power2.out'
    });
    
    // 进度圈发光
    if (progressRing) {
      gsap.to(progressRing, {
        filter: 'drop-shadow(0 0 10px #D4AF37)',
        duration: 0.3
      });
    }
    
    // 持续旋转动画
    gsap.to(backToTop, {
      rotation: '+=360',
      duration: 2,
      repeat: -1,
      ease: 'none'
    });
  });
  
  backToTop.addEventListener('mouseleave', () => {
    // 恢复大小
    gsap.to(backToTop, {
      scale: 1,
      duration: 0.3,
      ease: 'power2.in'
    });
    
    // 光晕减弱
    gsap.to(glow, {
      scale: 1,
      opacity: 0.5,
      duration: 0.4,
      ease: 'power2.in'
    });
    
    // 停止发光
    if (progressRing) {
      gsap.to(progressRing, {
        filter: 'none',
        duration: 0.3
      });
    }
    
    // 停止旋转
    gsap.killTweensOf(backToTop);
  });
  
  // 点击返回顶部
  backToTop.addEventListener('click', (e) => {
    e.preventDefault();
    
    // 点击波纹
    createRippleEffect(backToTop, e);
    
    // 震动反馈
    gsap.fromTo(backToTop,
      { x: -5 },
      {
        x: 5,
        duration: 0.05,
        repeat: 5,
        yoyo: true,
        ease: 'power1.inOut',
        onComplete: () => {
          gsap.to(backToTop, { x: 0, duration: 0.1 });
        }
      }
    );
    
    // 滚动到顶部
    gsap.to(window, {
      duration: 1.5,
      scrollTo: {
        y: 0,
        autoKill: false
      },
      ease: 'power3.inOut',
      onStart: () => {
        backToTop.classList.add('scrolling');
        
        // 光晕爆发
        gsap.to(glow, {
          scale: 2.5,
          opacity: 1,
          duration: 0.3,
          ease: 'power2.out',
          onComplete: () => {
            gsap.to(glow, {
              scale: 1.5,
              opacity: 0.5,
              duration: 0.5
            });
          }
        });
      },
      onUpdate: () => {
        // 实时更新进度圈
        const scrollY = window.scrollY;
        if (progressRing) {
          const scrollPercent = scrollY / (document.documentElement.scrollHeight - window.innerHeight);
          const offset = circumference - (scrollPercent * circumference);
          progressRing.style.strokeDashoffset = offset;
        }
      },
      onComplete: () => {
        backToTop.classList.remove('scrolling');
        
        // 完成动画
        gsap.fromTo(backToTop,
          { scale: 1.3 },
          {
            scale: 1,
            duration: 0.3,
            ease: 'elastic.out(1, 0.5)'
          }
        );
      }
    });
  });
}

/**
 * 波纹效果
 */
function createRippleEffect(element, event) {
  const ripple = document.createElement('span');
  ripple.className = 'ripple';
  
  const rect = element.getBoundingClientRect();
  const size = Math.max(rect.width, rect.height) * 2;
  
  ripple.style.cssText = `
    position: absolute;
    border-radius: 50%;
    background: radial-gradient(circle, rgba(212, 175, 55, 0.6), transparent);
    transform: translate(-50%, -50%);
    pointer-events: none;
    width: ${size}px;
    height: ${size}px;
    left: ${event.clientX - rect.left}px;
    top: ${event.clientY - rect.top}px;
  `;
  
  element.appendChild(ripple);
  
  gsap.fromTo(ripple,
    { scale: 0, opacity: 0.8 },
    {
      scale: 1,
      opacity: 0,
      duration: 0.8,
      ease: 'power2.out',
      onComplete: () => ripple.remove()
    }
  );
}

class ImageOptimizer {
  constructor() {
    this.observer = null;
    this.init();
  }
  
  init() {
    this.initLazyLoading();
    this.initResponsiveImages();
    this.initWebP();
    this.initImageCompression();
  }
  
  // 懒加载
  initLazyLoading() {
    const options = {
      rootMargin: '200px',
      threshold: 0.01
    };
    
    this.observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          this.loadImage(entry.target);
          this.observer.unobserve(entry.target);
        }
      });
    }, options);
    
    document.querySelectorAll('img[data-src]').forEach(img => {
      this.observer.observe(img);
    });
  }
  
  loadImage(img) {
    const src = img.dataset.src;
    const srcset = img.dataset.srcset;
    
    if (src) {
      img.src = src;
      delete img.dataset.src;
    }
    
    if (srcset) {
      img.srcset = srcset;
      delete img.dataset.srcset;
    }
    
    img.classList.add('loaded');
  }
  
  // 响应式图片
  initResponsiveImages() {
    const createResponsiveImage = (src, breakpoints) => {
      const picture = document.createElement('picture');
      
      breakpoints.forEach(({ width, type }) => {
        const source = document.createElement('source');
        source.srcset = `${src.replace(/\.[^.]+$/, `-${width}.${type}`)}`;
        source.type = `image/${type}`;
        source.media = `(max-width: ${width}px)`;
        picture.appendChild(source);
      });
      
      const img = document.createElement('img');
      img.src = src;
      img.loading = 'lazy';
      picture.appendChild(img);
      
      return picture;
    };
  }
  
  // WebP 检测
  initWebP() {
    const supportsWebP = () => {
      return document.createElement('canvas').toDataURL('image/webp').indexOf('data:image/webp') === 0;
    };
    
    if (supportsWebP()) {
      document.body.classList.add('webp');
    }
  }
