// App.tsx
import React, { useEffect, useRef } from 'react';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

import Footer from './components/Footer';

gsap.registerPlugin(ScrollTrigger);

const App: React.FC = () => {
  const mainRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    // 全局 ScrollTrigger 配置
    ScrollTrigger.config({
      limitCallbacks: true,
      ignoreMobileResize: true,

    return () => {
      // 清理所有 ScrollTrigger 实例
      ScrollTrigger.getAll().forEach(st => st.kill());
    };
  }, []);

};

export default App;

```tsx
// components/Navbar.tsx
import React, { useEffect, useRef, useState } from 'react';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

const Navbar: React.FC = () => {
  const navRef = useRef<HTMLElement>(null);
  const [activeSection, setActiveSection] = useState('hero');

  const navItems = [
    { id: 'hero', label: '首页' },
    { id: 'products', label: '产品' },
    { id: 'cases', label: '案例' },
    { id: 'clients', label: '客户' },
  ];

  const scrollToSection = (id: string) => {
    const element = document.getElementById(id);
    if (element) {
      gsap.to(window, {
        duration: 1,
        scrollTo: { y: element, offsetY: 80 },
        ease: 'power2.inOut',
      });
    }
  };

  useEffect(() => {
    // 使用 ScrollTrigger 监听滚动并更新导航高亮
    const sections = navItems.map(item => document.getElementById(item.id)).filter(Boolean) as HTMLElement[];

    sections.forEach((section) => {
      ScrollTrigger.create({
        trigger: section,
        start: 'top 30%',
        end: 'bottom 30%',
        onToggle: (self) => {
          if (self.isActive) {
            setActiveSection(section.id);
          }
        },
      });
    });

    // 导航栏背景变化
    ScrollTrigger.create({
      start: 'top -10%',
      onUpdate: (self) => {
        if (navRef.current) {
          if (self.progress > 0) {
            navRef.current.classList.remove('bg-transparent');
            navRef.current.classList.add('bg-black/80', 'backdrop-blur-md');
          } else {
            navRef.current.classList.add('bg-transparent');
            navRef.current.classList.remove('bg-black/80', 'backdrop-blur-md');
          }
        }
      },
    });

    return () => {
      ScrollTrigger.getAll().forEach(st => st.kill());
    };
  }, []);

  return (
    <nav ref={navRef} className="fixed top-0 left-0 right-0 z-50 transition-all duration-300 bg-transparent">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        <div className="flex items-center justify-between h-16">
          <div className="flex-shrink-0">
            <span className="text-2xl font-bold text-blue-400">AI 网络科技</span>
          </div>
          <div className="flex space-x-4">
            {navItems.map((item) => (
              <button
                key={item.id}
                onClick={() => scrollToSection(item.id)}
                className={`px-3 py-2 rounded-md text-sm font-medium transition-colors duration-200 ${
                  activeSection === item.id
                    ? 'text-blue-400 bg-blue-400/10'
                    : 'text-gray-300 hover:text-white hover:bg-white/10'
                }`}
              >
                {item.label}
              </button>
            ))}
          </div>
        </div>
      </div>
    </nav>
  );
};

export default Navbar;
```

| Hero (背景) | `top 100%` → `bottom 20%` | scrub: 1.5 | 条纹消隐、图片亮度、暗角 | 滚动驱动 |
| Hero (前景) | `top 80%` → `top 20%` | scrub: 1.5 + onEnter | 恒星文字诞生 | 滚动驱动 + 自动触发 |
| 产品 (图片) | `top 80%` → `center center` | scrub: 1.5 | 十字合拢、辉光 | 滚动驱动 |
| 产品 (文字) | `top 80%` → `top 20%` | scrub: 1.5 + onEnter | 涡流汇聚、3D旋转 | 滚动驱动 + 自动触发 |
| 案例 | `top 80%` → `top 20%` | scrub: 1.5 + toggleActions | 弹性坠落、倾斜滑入 | 滚动驱动 + 自动触发 |
| 客户 | `top 85%` → `top 25%` | scrub: 1.5 + toggleActions | 递进滑入、遮罩渐显 | 滚动驱动 + 自动触发 |

### 5.3 性能优化建议
- **GPU加速**: 动画属性优先使用 `transform` 和 `opacity`，这些属性可以触发 GPU 合成，减少重绘和回流。
- **will-change**: 对动画元素应用 `will-change: transform, opacity` 属性，提前告知浏览器哪些属性会变化，以便进行优化。
- **Canvas离屏渲染**: 对于 `CardTextCorona` 和 `CardTextFluid3D` 中的粒子/星空背景，使用离屏 Canvas 进行渲染，避免阻塞主线程。
- **IntersectionObserver懒加载**: 对于非首屏的组件，可以使用 IntersectionObserver 实现懒加载，当组件进入视口时再初始化其内部的 GSAP 动画和 Canvas 渲染。
- **requestAnimationFrame**: 对于持续的动画循环（如 `CardTextCorona` 的粒子动画和 `CardTextFluid3D` 的涡流动画），使用 `requestAnimationFrame` 代替 `setInterval`，以获得更平滑的动画和更好的性能。
- **GSAP Context**: 使用 `gsap.context()` 包裹所有动画创建，确保在组件卸载时能一次性清除所有 GSAP 动画和 ScrollTrigger 实例，避免内存泄漏。

### 6.3 响应式断点策略

| `sm` | 640px | 调整卡片布局为单列或两列 |
