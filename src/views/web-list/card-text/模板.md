## 3. Props 定义

```typescript
interface Props {
  // 必填属性
  propName: string              // 属性说明

  // 可选属性（带默认值）
  optionalProp?: string        // 属性说明，默认值：'default'
  numericProp?: number          // 属性说明，默认值：0
  booleanProp?: boolean         // 属性说明，默认值：false

  // 数组/对象
  listProp?: string[]           // 属性说明，默认值：[]
  objectProp?: Record<string, unknown> // 属性说明
}

// TypeScript withDefaults
const props = withDefaults(defineProps<Props>(), {
  optionalProp: 'default',
  numericProp: 0,
  booleanProp: false,
  listProp: () => [],
  objectProp: () => ({})
})
```

## 8. 技术实现要点

### 8.1 ref 获取 DOM

```typescript
// 单元素
const containerRef = ref<HTMLElement | null>(null)
const titleRef = ref<HTMLElement | null>(null)

// 多元素 (v-for)
const itemRefs = ref<(HTMLElement | null)[]>([])
const itemSet = new Set<HTMLElement>()

const setItemRef = (el: HTMLElement | null, index: number) => {
  if (el) {
    itemRefs.value[index] = el
    itemSet.add(el)
  }
}
```

### 8.2 gsap.context 隔离

```typescript
let ctx: gsap.Context | null = null

onMounted(() => {
  const rootEl = containerRef.value
  if (!rootEl) return

  ctx = gsap.context(() => {
    // 所有动画定义
  }, rootEl)
})
```

### 8.3 清理机制

```typescript
// 类型别名
type TweenCleanup = () => void
const cleanupFns: TweenCleanup[] = []

// 注册清理
cleanupFns.push(() => tl.kill())

// 卸载清理
onUnmounted(() => {
  cleanupFns.forEach(fn => fn())
  cleanupFns.length = 0
  ctx?.revert()
  ctx = null
  ScrollTrigger.getAll().forEach(st => st.kill())
})
```

## 12. API 参考

### Props

| 属性名 | 类型 | 默认值 | 说明 |
|--------|------|--------|------|
| title | string | - | 标题（必填） |
| subtitle | string | '' | 副标题 |
| list | array | [] | 列表数据 |
| duration | number | 1 | 动画时长 |
| theme | 'gradient' \| 'dark' \| 'light' | 'gradient' | 主题色 |

### Events

| 事件名 | 参数 | 说明 |
|--------|------|------|
| click | (id: number) | 点击事件 |
| change | (value: string) | 变更事件 |

### Slots

| 插槽名 | 说明 |
|--------|------|
| default | 默认插槽 |
| header | 头部插槽 |
| footer | 底部插槽 |
